def translate(text: str) -> str:
    # Edge Case: Empty string input
    if not text:
        return ""
    
    words = text.split()
    translated_words = []
    
    vowels = set('aeiou')
    
    for word in words:
        # Edge Case: Empty word (multiple spaces)
        if not word:
            translated_words.append(word)
            continue
            
        # Edge Case: Single character word
        if len(word) == 1:
            if word in vowels:
                translated_words.append(word + 'ay')
            else:
                translated_words.append(word + 'ay')
            continue
            
        # Rule 1: If word begins with a vowel, or starts with "xr" or "yt"
        if word[0] in vowels or word.startswith('xr') or word.startswith('yt'):
            translated_words.append(word + 'ay')
            continue
            
        # Find the index where the consonant cluster ends
        i = 0
        while i < len(word):
            # Edge Case: Handle 'qu' as a unit when it follows consonants
            if i < len(word) - 1 and word[i:i+2] == 'qu':
                i += 2
                break
            # Edge Case: Handle 'y' as a vowel when it follows consonants
            elif word[i] == 'y' and i > 0:
                break
            # Normal consonant
            elif word[i] not in vowels:
                i += 1
            else:
                break
                
        # Apply transformation based on the rules
        consonant_cluster = word[:i]
        rest_of_word = word[i:]
        translated_words.append(rest_of_word + consonant_cluster + 'ay')
    
    return ' '.join(translated_words)

export const compute = (strandA, strandB) => {
  // Edge Case: Handle undefined or null inputs
  if (strandA === undefined || strandB === undefined) {
    throw new Error('strands must be supplied');
  }
  // Edge Case: Handle non-string inputs by converting to string
  if (typeof strandA !== 'string') strandA = String(strandA);
  if (typeof strandB !== 'string') strandB = String(strandB);
  // Edge Case: Handle strands of different lengths
  if (strandA.length !== strandB.length) {
    throw new Error('strands must be of equal length');
  }
  // Edge Case: Handle invalid characters in strands
  const validChars = /^[CAGT]*$/;
  if (!validChars.test(strandA) || !validChars.test(strandB)) {
    throw new Error('strands must contain only C, A, G, or T');
  }
  // Edge Case: Handle empty strands
  if (strandA.length === 0) {
    return 0;
  }
  let distance = 0;
  for (let i = 0; i < strandA.length; i++) {
    if (strandA[i] !== strandB[i]) {
      distance++;
    }
  }
  return distance;
};

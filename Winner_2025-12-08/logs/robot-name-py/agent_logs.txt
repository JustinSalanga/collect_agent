import string
class Robot:
    _used_names = set()
    def __init__(self):
        self._name = None
        self._generate_name()
    @property
    def name(self) -> str:
        return self._name
    def reset(self) -> None:
        # Edge Case: Removing the current name from used names before generating a new one
        if self._name:
            self._used_names.discard(self._name)
        self._generate_name()
    def _generate_name(self) -> None:
        # Edge Case: Handling name collision by retrying until a unique name is found
        # Prevent infinite loop by limiting attempts
        max_attempts = 1000
        attempts = 0
        
        while attempts < max_attempts:
            # Generate two random uppercase letters
            letters = ''.join(random.choices(string.ascii_uppercase, k=2))
            # Generate three random digits
            digits = ''.join(random.choices(string.digits, k=3))
            # Combine to form the name
            new_name = letters + digits
            
            # Check if this name is already used
            if new_name not in self._used_names:
                self._name = new_name
                self._used_names.add(new_name)
                return
            attempts += 1
        
        # If we've exhausted our attempts, raise an exception
        raise RuntimeError("Unable to generate unique robot name after maximum attempts")

// This is only a SKELETON file for the 'Resistor Color Trio' exercise. It's been provided as a
// convenience to get you started writing code faster.
//
export class ResistorColorTrio {
  constructor(colors) {
    if (!colors || colors.length < 3) {
      throw new Error('At least three colors need to be present');
    }
    
    this.colors = colors;
    this.colorValues = {
      'black': 0,
      'brown': 1,
      'red': 2,
      'orange': 3,
      'yellow': 4,
      'green': 5,
      'blue': 6,
      'violet': 7,
      'grey': 8,
      'white': 9
    };
  }
  label() {
    // Edge Case: Invalid color input
    for (let i = 0; i < 3; i++) {
      if (!this.colorValues.hasOwnProperty(this.colors[i].toLowerCase())) {
        throw new Error(`Invalid color: ${this.colors[i]}`);
      }
    }
    
    const firstValue = this.colorValues[this.colors[0].toLowerCase()];
    const secondValue = this.colorValues[this.colors[1].toLowerCase()];
    const zeros = this.colorValues[this.colors[2].toLowerCase()];
    
    // Edge Case: Calculate main value
    const mainValue = firstValue * 10 + secondValue;
    
    // Edge Case: Calculate total ohms value
    let ohmsValue = mainValue;
    // Added iteration limit to prevent potential infinite loop
    const maxIterations = 15; // Reasonable limit for resistor color codes
    for (let i = 0; i < Math.min(zeros, maxIterations); i++) {
      ohmsValue *= 10;
    }
    
    // Edge Case: Zero value result
    if (ohmsValue === 0) {
      return "0 ohms";
    }
    
    // Edge Case: Apply metric prefixes for large values
    if (ohmsValue >= 1000000000000) {
      return `${ohmsValue / 1000000000000} teraohms`;
    } else if (ohmsValue >= 1000000000) {
      return `${ohmsValue / 1000000000} gigaohms`;
    } else if (ohmsValue >= 1000000) {
      return `${ohmsValue / 1000000} megaohms`;
    } else if (ohmsValue >= 1000) {
      return `${ohmsValue / 1000} kiloohms`;
    } else {
      return `${ohmsValue} ohms`;
    }
  }
}

// This is only a SKELETON file for the 'Robot Name' exercise. It's been
// provided as a convenience to get your started writing code faster.
const usedNames = new Set();
export class Robot {
    constructor() {
        this._name = this.generateName();
    }
    /**
     * @return {string}
     */
    get name() {
        return this._name;
    }
    /**
     * @return {void}
     */
    reset() {
        // Edge Case: Remove old name from used names set before generating new one
        if (this._name) {
            usedNames.delete(this._name);
        }
        this._name = this.generateName();
    }
    generateName() {
        let name;
        let attempts = 0;
        const maxAttempts = 1000000; // Prevent infinite loop
        do {
            // Edge Case: Handle potential infinite loop when generating names
            if (attempts > maxAttempts) {
                throw new Error('Unable to generate unique name - too many attempts');
            }
            const letters = String.fromCharCode(
                Math.floor(Math.random() * 26) + 65,
                Math.floor(Math.random() * 26) + 65
            );
            const numbers = Math.floor(Math.random() * 1000)
                .toString()
                .padStart(3, '0');
            name = letters + numbers;
            attempts++;
        } while (usedNames.has(name));
        // Edge Case: Add newly generated name to used names set
        usedNames.add(name);
        return name;
    }
    static releaseNames() {
        // Edge Case: Clear all used names when releasing names
        usedNames.clear();
    }
}

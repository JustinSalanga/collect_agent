export class List {
  constructor(values = []) {
    this.values = values;
  }
  compare(other) {
    // Edge Case: Both lists are empty
    if (this.values.length === 0 && other.values.length === 0) {
      return 'EQUAL';
    }
    
    // Edge Case: First list is empty, second is not
    if (this.values.length === 0) {
      return 'SUBLIST';
    }
    
    // Edge Case: Second list is empty, first is not
    if (other.values.length === 0) {
      return 'SUPERLIST';
    }
    
    // Edge Case: Lists have same length, check for equality
    if (this.values.length === other.values.length) {
      for (let i = 0; i < this.values.length; i++) {
        if (this.values[i] !== other.values[i]) {
          return 'UNEQUAL';
        }
      }
      return 'EQUAL';
    }
    
    // Edge Case: First list is shorter than second, check if it's a sublist
    if (this.values.length < other.values.length) {
      return this.isSublist(this.values, other.values) ? 'SUBLIST' : 'UNEQUAL';
    }
    
    // Edge Case: First list is longer than second, check if it's a superlist
    return this.isSublist(other.values, this.values) ? 'SUPERLIST' : 'UNEQUAL';
  }
  
  // Helper function to check if smallerList is a contiguous sublist of largerList
  isSublist(smallerList, largerList) {
    // Edge Case: Empty list is always a sublist
    if (smallerList.length === 0) {
      return true;
    }
    
    // Edge Case: Smaller list is actually larger
    if (smallerList.length > largerList.length) {
      return false;
    }
    
    // Check for contiguous subsequence
    for (let i = 0; i <= largerList.length - smallerList.length; i++) {
      let found = true;
      for (let j = 0; j < smallerList.length; j++) {
        if (largerList[i + j] !== smallerList[j]) {
          found = false;
          break;
        }
      }
      if (found) {
        return true;
      }
    }
    return false;
  }
}

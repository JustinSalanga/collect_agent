const usedNames = new Set();
export class Robot {
    constructor() {
        this._name = this.generateName();
    }
    /**
     * @return {string}
     */
    get name() {
        return this._name;
    }
    /**
     * @return {void}
     */
    reset() {
        // Edge Case: Remove old name from used names set before generating new one
        if (this._name) {
            usedNames.delete(this._name);
        }
        
        this._name = this.generateName();
    }
    generateName() {
        // Edge Case: Handle case when all possible names are exhausted
        if (usedNames.size >= 26 * 26 * 1000) {
            throw new Error('All possible names have been used');
        }
        
        let name;
        let attempts = 0;
        const maxAttempts = 1000; // Prevent infinite loop
        
        do {
            const letters = String.fromCharCode(
                Math.floor(Math.random() * 26) + 65,
                Math.floor(Math.random() * 26) + 65
            );
            const numbers = String(Math.floor(Math.random() * 1000)).padStart(3, '0');
            name = letters + numbers;
            attempts++;
            
            // Prevent infinite loop by capping attempts
            if (attempts >= maxAttempts) {
                throw new Error('Unable to generate unique name after maximum attempts');
            }
        } while (usedNames.has(name));
        
        usedNames.add(name);
        return name;
    }
}

import random
import string
class Robot:
    _used_names = set()
    def __init__(self):
        self._name = None
        self._generate_name()
    @property
    def name(self) -> str:
        return self._name
    def reset(self) -> None:
        # Edge Case: Resetting a robot releases its name for reuse
        if self._name:
            self._used_names.discard(self._name)
        self._name = None
        self._generate_name()
    def _generate_name(self) -> None:
        # Edge Case: Handle name generation when robot already has a name
        if self._name is not None:
            return
        # Edge Case: Exhausting all possible name combinations (26*26*10*10*10 = 676,000 names)
        max_attempts = 1000  # Prevent infinite loop in case of high collision rate
        attempts = 0
        while attempts < max_attempts:
            # Generate random name with two uppercase letters followed by three digits
            letters = ''.join(random.choices(string.ascii_uppercase, k=2))
            digits = ''.join(random.choices(string.digits, k=3))
            new_name = letters + digits
            # Edge Case: Name collision after generating a random name
            if new_name not in self._used_names:
                self._name = new_name
                self._used_names.add(new_name)
                return
            attempts += 1
        # Edge Case: All possible names are exhausted
        raise RuntimeError("All possible robot names have been exhausted")

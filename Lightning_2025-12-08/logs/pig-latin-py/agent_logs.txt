def translate(text: str) -> str:
    if not text:
        return ""
    
    words = text.split()
    translated_words = []
    
    vowels = set('aeiou')
    
    for word in words:
        # Edge Case: empty string
        if not word:
            translated_words.append("")
            continue
            
        # Edge Case: word starts with vowel or special cases "xr" or "yt"
        if word[0] in vowels or word.startswith(('xr', 'yt')):
            translated_words.append(word + 'ay')
            continue
        
        # Find the index where the consonant cluster ends
        i = 0
        while i < len(word):
            # Edge Case: handle "qu" as a unit
            if i < len(word) - 1 and word[i:i+2] == 'qu':
                i += 2
                break
            # Edge Case: handle 'y' as a vowel when it's not at the beginning
            elif word[i] == 'y' and i > 0:
                break
            # Edge Case: stop at vowels
            elif word[i] in vowels:
                break
            i += 1
        
        # Edge Case: word with no vowels (treat entire word as consonants)
        if i == len(word):
            translated_words.append(word + 'ay')
        else:
            translated_words.append(word[i:] + word[:i] + 'ay')
    
    return ' '.join(translated_words)

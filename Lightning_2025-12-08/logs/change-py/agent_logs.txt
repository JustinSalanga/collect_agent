def find_fewest_coins(coins, target):
    # Edge Case: Negative target amount
    if target < 0:
        raise ValueError("target can't be negative")
    
    # Edge Case: Zero target amount
    if target == 0:
        return []
    
    # Edge Case: Empty coin list
    if not coins:
        raise ValueError("can't make target with given coins")
    
    # Edge Case: Coins with zero or negative values
    if any(coin <= 0 for coin in coins):
        raise ValueError("coins must be positive")
    
    # Sort coins in descending order for greedy approach consideration,
    # but we'll use dynamic programming which works regardless of order
    coins = [coin for coin in coins if coin > 0]  # Filter out any non-positive coins
    
    # Initialize dp array: dp[i] will hold the minimum number of coins needed for amount i
    dp = [float('inf')] * (target + 1)
    dp[0] = 0
    
    # Initialize parent array to reconstruct the solution
    parent = [-1] * (target + 1)
    
    # Fill dp array
    for i in range(1, target + 1):
        for coin in coins:
            if coin <= i and dp[i - coin] + 1 < dp[i]:
                dp[i] = dp[i - coin] + 1
                parent[i] = coin
    
    # Edge Case: Target amount cannot be formed with given coins
    if dp[target] == float('inf'):
        raise ValueError("can't make target with given coins")
    
    # Reconstruct the solution
    result = []
    current = target
    while current > 0:
        coin = parent[current]
        if coin == -1:  # Safety check to prevent infinite loop
            raise ValueError("can't make target with given coins")
        result.append(coin)
        current -= coin
    
    # Return coins in ascending order
    return sorted(result)

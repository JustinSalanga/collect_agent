export const toDecimal = (hexString) => {
  // Edge Case: Empty string input
  if (!hexString || hexString.length === 0) {
    return 0;
  }
  // Edge Case: String with whitespace
  hexString = hexString.trim();
  // Edge Case: Handle '0x' or '0X' prefix
  if (hexString.toLowerCase().startsWith('0x')) {
    hexString = hexString.substring(2);
  }
  // Edge Case: Empty string after trimming or removing prefix
  if (hexString.length === 0) {
    return 0;
  }
  let result = 0;
  const validChars = '0123456789abcdefABCDEF';
  // Added iteration limit as a safety measure
  const maxIterations = 1000;
  for (let i = 0; i < hexString.length && i < maxIterations; i++) {
    const char = hexString[i];
    
    // Edge Case: Invalid characters in string
    if (!validChars.includes(char)) {
      return 0;
    }
    // Convert character to its decimal value
    let value;
    if (char >= '0' && char <= '9') {
      value = char.charCodeAt(0) - '0'.charCodeAt(0);
    } else if (char >= 'a' && char <= 'f') {
      value = char.charCodeAt(0) - 'a'.charCodeAt(0) + 10;
    } else { // char >= 'A' && char <= 'F'
      value = char.charCodeAt(0) - 'A'.charCodeAt(0) + 10;
    }
    result = result * 16 + value;
  }
  return result;
};

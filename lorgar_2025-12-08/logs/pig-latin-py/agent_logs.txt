def translate(text: str) -> str:
    if not text:
        return ""
    
    words = text.split()
    translated_words = []
    
    vowels = set('aeiou')
    
    for word in words:
        # Edge Case: empty string
        if not word:
            translated_words.append("")
            continue
            
        # Edge Case: single character word
        if len(word) == 1:
            if word in vowels:
                translated_words.append(word + "ay")
            else:
                translated_words.append(word + "ay")
            continue
            
        # Rule 1: If word begins with a vowel, or starts with "xr" or "yt"
        if word[0] in vowels or word.startswith("xr") or word.startswith("yt"):
            translated_words.append(word + "ay")
            continue
            
        # Find the index where the consonant cluster ends
        i = 0
        while i < len(word):
            # Edge Case: handle "qu" as a unit
            if i < len(word) - 1 and word[i:i+2] == "qu":
                i += 2
                break
            # Edge Case: handle "y" as a vowel when it's not the first letter
            elif word[i] == 'y' and i > 0:
                break
            # If we hit a vowel (including y when not at the start), stop
            elif word[i] in vowels:
                break
            i += 1
            
        # Move the consonant cluster (and possibly "qu") to the end and add "ay"
        translated_word = word[i:] + word[:i] + "ay"
        translated_words.append(translated_word)
        
    return " ".join(translated_words)

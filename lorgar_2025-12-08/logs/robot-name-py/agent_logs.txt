import random
import string
class Robot:
    _used_names = set()
    def __init__(self):
        self._name = None
        self._generate_name()
    @property
    def name(self) -> str:
        return self._name
    def reset(self) -> None:
        # Edge Case: Resetting a robot's name removes the old name from used names
        if self._name:
            self._used_names.discard(self._name)
        self._name = None
        self._generate_name()
    def _generate_name(self) -> None:
        # Edge Case: Handle name generation when many names are already used
        max_attempts = 1000000  # Prevent infinite loop in case of exhaustion
        attempts = 0
        
        while attempts < max_attempts:
            # Generate random name with 2 uppercase letters and 3 digits
            letters = ''.join(random.choices(string.ascii_uppercase, k=2))
            digits = ''.join(random.choices(string.digits, k=3))
            new_name = letters + digits
            
            # Edge Case: Ensure generated name is unique
            if new_name not in self._used_names:
                self._used_names.add(new_name)
                self._name = new_name
                return
            
            attempts += 1
        
        # Edge Case: Raise exception if name space is exhausted
        raise RuntimeError("Unable to generate unique robot name - name space exhausted")

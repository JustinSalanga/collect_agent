// This is only a SKELETON file for the 'Robot Name' exercise. It's been
// provided as a convenience to get your started writing code faster.
const usedNames = new Set();
export class Robot {
    constructor() {
        this._name = this.generateName();
    }
    /**
     * @return {string}
     */
    get name() {
        return this._name;
    }
    /**
     * @return {void}
     */
    reset() {
        // Edge Case: Remove old name from used names set before generating new name
        if (this._name) {
            usedNames.delete(this._name);
        }
        
        // Edge Case: Generate a new unique name after reset
        this._name = this.generateName();
    }
    generateName() {
        // Edge Case: Handle case when all possible names are exhausted (though extremely unlikely)
        if (usedNames.size >= 26 * 26 * 10 * 10 * 10) {
            throw new Error("All possible names have been used");
        }
        
        let name;
        let attempts = 0;
        const maxAttempts = 10000; // Prevent infinite loop
        
        do {
            const letters = String.fromCharCode(
                Math.floor(Math.random() * 26) + 65,
                Math.floor(Math.random() * 26) + 65
            );
            const numbers = Math.floor(Math.random() * 1000).toString().padStart(3, '0');
            name = letters + numbers;
            attempts++;
            
            // Prevent infinite loop by capping attempts
            if (attempts >= maxAttempts) {
                throw new Error("Unable to generate unique name after maximum attempts");
            }
        } while (usedNames.has(name));
        
        usedNames.add(name);
        return name;
    }
    static releaseNames() {
        // Edge Case: Clear all used names to allow reuse after factory reset of all robots
        usedNames.clear();
    }
}

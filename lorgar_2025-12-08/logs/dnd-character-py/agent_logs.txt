class Character:
    def __init__(self):
        self.strength = self.ability()
        self.dexterity = self.ability()
        self.constitution = self.ability()
        self.intelligence = self.ability()
        self.wisdom = self.ability()

    def ability(self):

        dice = [random.randint(1, 6) for _ in range(4)]

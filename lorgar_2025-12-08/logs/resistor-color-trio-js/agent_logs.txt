// This is only a SKELETON file for the 'Resistor Color Trio' exercise. It's been provided as a

export class ResistorColorTrio {
  constructor(colors) {
    if (colors.length < 3) {
      throw new Error('At least three colors need to be present');
    }
    this.colors = colors;
    this.colorValues = {
      'black': 0,
      'brown': 1,
      'red': 2,
      'orange': 3,
      'yellow': 4,
      'green': 5,
      'blue': 6,
      'violet': 7,
      'grey': 8,
      'white': 9
    };
  }
  label() {
    // Edge Case: Invalid color inputs
    for (let i = 0; i < 3; i++) {
      if (!(this.colors[i] in this.colorValues)) {
        throw new Error('Invalid color: ' + this.colors[i]);
      }
    }
    const firstValue = this.colorValues[this.colors[0]];
    const secondValue = this.colorValues[this.colors[1]];
    const thirdValue = this.colorValues[this.colors[2]];
    // Edge Case: Zero value (black-black-black)
    if (firstValue === 0 && secondValue === 0 && thirdValue === 0) {
      return "0 ohms";
    }
    const mainValue = firstValue * 10 + secondValue;
    const totalValue = mainValue * Math.pow(10, thirdValue);
    // Edge Case: Exact prefix boundaries (e.g., 1000, 1000000)
    if (totalValue >= 1000000000 && totalValue % 1000000000 === 0) {
      return (totalValue / 1000000000) + " gigaohms";
    } else if (totalValue >= 1000000 && totalValue % 1000000 === 0) {
      return (totalValue / 1000000) + " megaohms";
    } else if (totalValue >= 1000 && totalValue % 1000 === 0) {
      return (totalValue / 1000) + " kiloohms";
    } else {
      return totalValue + " ohms";
    }
  }
}

    def __init__(self, matrix_string):
        # Edge Case: Empty string input
        if not matrix_string:
            self.matrix = []
            return
            
        # Split the input string into lines and convert to a 2D list of integers
        lines = matrix_string.strip().split('
')
        self.matrix = []
        
        # Edge Case: Handle empty lines
        for line in lines:
            if line.strip():  # Only process non-empty lines
                row = [int(num) for num in line.split()]
                self.matrix.append(row)
        
        # Edge Case: Non-rectangular matrix (unequal row lengths)
        # We'll handle this by returning only complete columns up to the shortest row length
        if self.matrix:
            self.num_rows = len(self.matrix)
            self.num_cols = min(len(row) for row in self.matrix) if self.matrix else 0
        else:
            self.num_rows = 0
            self.num_cols = 0
    def row(self, index):
        # Edge Case: Index out of bounds (less than 1 or greater than number of rows)
        if index < 1 or index > self.num_rows:
            raise IndexError("Index out of range")
        
        # Edge Case: Empty matrix
        if not self.matrix:
            return []
            
        # Return a copy of the requested row
        return self.matrix[index-1].copy()
    
    def column(self, index):
        # Edge Case: Index out of bounds (less than 1 or greater than number of columns)
        if index < 1 or index > self.num_cols:
            raise IndexError("Index out of range")
        
        # Edge Case: Empty matrix
        if not self.matrix:
            return []
            
        # Extract the column values
        return [self.matrix[row][index-1] for row in range(self.num_rows)]

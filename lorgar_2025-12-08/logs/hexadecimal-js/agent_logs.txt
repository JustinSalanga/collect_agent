export const toDecimal = (hex) => {
  // Edge Case: Empty string input
  if (!hex || hex.length === 0) {
    return 0;
  }
  // Edge Case: Invalid characters in hexadecimal string
  const validChars = /^[0-9a-fA-F]+$/;
  if (!validChars.test(hex)) {
    return 0;
  }
  let result = 0;
  const lowerHex = hex.toLowerCase();
  
  // Edge Case: Very large hexadecimal strings
  // Using first principles - processing each digit from left to right
  for (let i = 0; i < lowerHex.length; i++) {
    const char = lowerHex[i];
    let digitValue;
    
    if (char >= '0' && char <= '9') {
      digitValue = char.charCodeAt(0) - '0'.charCodeAt(0);
    } else if (char >= 'a' && char <= 'f') {
      digitValue = char.charCodeAt(0) - 'a'.charCodeAt(0) + 10;
    }
    
    // Edge Case: Potential overflow - but we continue with the calculation as per requirements
    result = result * 16 + digitValue;
  }
  
  return result;
};

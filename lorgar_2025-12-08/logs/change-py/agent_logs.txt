def find_fewest_coins(coins, target):
    # Edge Case: Negative target amount
    if target < 0:
        raise ValueError("target can't be negative")
    
    # Edge Case: Zero target amount
    if target == 0:
        return []
    
    # Edge Case: Empty coin list
    if not coins:
        raise ValueError("can't make target with given coins")
    
    # Edge Case: Coins include zero or negative values
    if any(coin <= 0 for coin in coins):
        raise ValueError("coins must be positive")
    
    # Sort coins in descending order for optimization
    sorted_coins = sorted(coins, reverse=True)
    
    # Initialize dp array: dp[i] represents the minimum number of coins needed for amount i
    dp = [float('inf')] * (target + 1)
    dp[0] = 0
    
    # Keep track of which coin was used to achieve the minimum for each amount
    coin_used = [None] * (target + 1)
    
    # Fill dp array
    for i in range(1, target + 1):
        for coin in sorted_coins:
            # Edge Case: Coin value larger than current amount
            if coin > i:
                continue
            if dp[i - coin] + 1 < dp[i]:
                dp[i] = dp[i - coin] + 1
                coin_used[i] = coin
    
    # Edge Case: Target amount cannot be formed with given coins
    if dp[target] == float('inf'):
        raise ValueError("can't make target with given coins")
    
    # Reconstruct the solution
    result = []
    remaining = target
    while remaining > 0:
        coin = coin_used[remaining]
        result.append(coin)
        remaining -= coin
    
    # Return coins in ascending order
    return sorted(result)

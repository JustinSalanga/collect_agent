def is_isogram(string: str) -> bool:
    # Edge Case: Empty string is considered an isogram
    if not string:
        return True
    
    # Convert to lowercase for case-insensitive comparison
    lowercase_string = string.lower()
    
    # Keep track of seen letters
    seen_letters = set()
    
    # Iterate through each character in the string
    for char in lowercase_string:
        # Edge Case: Ignore spaces and hyphens
        if char == ' ' or char == '-':
            continue
            
        # Edge Case: If character is not alphabetic, it's still checked for repetition
        if char in seen_letters:
            return False
        seen_letters.add(char)
    
    return True
# Handled Edge Cases: empty string, case sensitivity, spaces, hyphens, non-alphabetic characters, single character strings

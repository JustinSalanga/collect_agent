/**
 * @param {number[][]} matrix
 * @returns {{row: number, column: number}[]}
 */
export const saddlePoints = (matrix) => {
  // Edge Case: Empty matrix or matrix with no columns
  if (!matrix || matrix.length === 0 || matrix[0].length === 0) {
    return [];
  }
  const rows = matrix.length;
  const cols = matrix[0].length;
  const result = [];
  // Precompute row maxima
  const rowMax = new Array(rows).fill(Number.NEGATIVE_INFINITY);
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      if (matrix[i][j] > rowMax[i]) {
        rowMax[i] = matrix[i][j];
      }
    }
  }
  // Precompute column minima
  const colMin = new Array(cols).fill(Number.POSITIVE_INFINITY);
  for (let j = 0; j < cols; j++) {
    for (let i = 0; i < rows; i++) {
      if (matrix[i][j] < colMin[j]) {
        colMin[j] = matrix[i][j];
      }
    }
  }
  // Find saddle points
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      // Edge Case: Value is both maximum in its row and minimum in its column
      if (matrix[i][j] === rowMax[i] && matrix[i][j] === colMin[j]) {
        result.push({ row: i + 1, column: j + 1 }); // 1-indexed
      }
    }
  }
  return result;
};

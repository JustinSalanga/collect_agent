    # Edge Case: Negative number
    if number < 0:
        raise ValueError("Number must be non-negative")
    
    # Edge Case: Zero
    if number == 0:
        return 0
    
    # Using formula: (n(n+1)/2)^2
    total = number * (number + 1) // 2
    return total * total
def sum_of_squares(number: int) -> int:
    # Edge Case: Negative number

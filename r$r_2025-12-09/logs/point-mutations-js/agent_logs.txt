export class DNA {
  /**
   * @param {string} nucleotides
   */
  constructor(nucleotides) {
    this.nucleotides = nucleotides;
  }
  /**
   * @param {string} comparison
   * @returns {number}
   */
  hammingDistance(comparison) {
    // Edge Case: Handle null or undefined inputs
    if (comparison === null || comparison === undefined) {
      throw new Error('Invalid comparison string');
    }
    // Edge Case: Handle non-string inputs by converting to string
    if (typeof comparison !== 'string') {
      comparison = String(comparison);
    }
    const strandA = this.nucleotides;
    const strandB = comparison;
    
    // Edge Case: Handle empty strings
    if (strandA.length === 0 && strandB.length === 0) {
      return 0;
    }
    // Edge Case: According to problem statement, we can assume equal length sequences
    // But we'll still handle the case where they might not be equal by taking the minimum length
    const minLength = Math.min(strandA.length, strandB.length);
    
    let distance = 0;
    
    // Edge Case: Handle single character strands
    for (let i = 0; i < minLength; i++) {
      if (strandA[i] !== strandB[i]) {
        distance++;
      }
    }
    
    return distance;
  }
}

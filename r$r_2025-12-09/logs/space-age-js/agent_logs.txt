/**
 * @param {string} planet
 * @param {number} seconds
 * @returns {number}
 */
export const age = (planet, seconds) => {
  // Edge Case: Negative or zero age in seconds
  if (seconds <= 0) {
    throw new Error('Age in seconds must be positive');
  }
  // Edge Case: Invalid planet name input (case sensitive)
  const planetPeriods = {
    'Mercury': 0.2408467,
    'Venus': 0.61519726,
    'Earth': 1.0,
    'Mars': 1.8808158,
    'Jupiter': 11.862615,
    'Saturn': 29.447498,
    'Uranus': 84.016846,
    'Neptune': 164.79132
  };
  // Edge Case: Planet name not in the list
  if (!planetPeriods.hasOwnProperty(planet)) {
    throw new Error('not a planet');
  }
  // Calculate Earth years
  const earthYears = seconds / 31557600;
  
  // Calculate age on the specified planet
  const planetAge = earthYears / planetPeriods[planet];
  
  // Round to two decimal places
  return Math.round(planetAge * 100) / 100;
};

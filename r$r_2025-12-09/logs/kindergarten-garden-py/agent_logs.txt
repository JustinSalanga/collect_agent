class Garden:
    def __init__(self, diagram: str, students: list[str] | None = None) -> None:
        # Edge Case: Handle empty diagram
        if not diagram:
            self.rows = ["", ""]
        else:
            # Split diagram into rows
            lines = diagram.split('
')
            # Edge Case: Handle diagram with only one row
            if len(lines) == 1:
                self.rows = [lines[0], ""]
            else:
                self.rows = [lines[0], lines[1]]
        
        # Edge Case: Handle mismatched row lengths by taking minimum length
        self.row_length = min(len(self.rows[0]), len(self.rows[1]))
        
        # Edge Case: Handle custom student list
        if students is not None:
            self.students = sorted(students)
        else:
            # Default student roster
            self.students = ["Alice", "Bob", "Charlie", "David", "Eve", "Fred",
                           "Ginny", "Harriet", "Ileana", "Joseph", "Kincaid", "Larry"]
        
        # Create mapping from plant codes to full names
        self.plant_map = {
            'G': 'Grass',
            'C': 'Clover',
            'R': 'Radishes',
            'V': 'Violets'
        }
    
    def plants(self, student: str) -> list[str]:
        # Edge Case: Handle student not in roster
        if student not in self.students:
            return []
        
        # Find student's position
        student_index = self.students.index(student)
        
        # Each student gets 2 consecutive cups in each row, starting at position 2*student_index
        start_pos = 2 * student_index
        
        # Edge Case: Handle garden with fewer cups than students need
        if start_pos + 1 >= self.row_length:
            # If there aren't enough cups, return what we can
            result = []
            if start_pos < len(self.rows[0]):
                result.append(self.plant_map.get(self.rows[0][start_pos], ""))
            if start_pos + 1 < len(self.rows[0]):
                result.append(self.plant_map.get(self.rows[0][start_pos + 1], ""))
            if start_pos < len(self.rows[1]):
                result.append(self.plant_map.get(self.rows[1][start_pos], ""))
            if start_pos + 1 < len(self.rows[1]):
                result.append(self.plant_map.get(self.rows[1][start_pos + 1], ""))
            # Filter out empty strings
            return [plant for plant in result if plant]
        
        # Get the four plants (two from each row)
        plant_codes = [
            self.rows[0][start_pos],
            self.rows[0][start_pos + 1],
            self.rows[1][start_pos],
            self.rows[1][start_pos + 1]
        ]
        
        # Convert codes to full plant names
        result = [self.plant_map[code] for code in plant_codes]
        
        return result

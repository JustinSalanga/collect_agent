def parse_binary(binary_string: str) -> int:
    # Edge Case: Input is None
    if binary_string is None:
        raise ValueError("Invalid binary literal: None")
    
    # Edge Case: Input is not a string
    if not isinstance(binary_string, str):
        raise ValueError("Invalid binary literal: " + str(binary_string))
    
    # Edge Case: Empty string
    if binary_string == "":
        raise ValueError("Invalid binary literal: ")
    
    # Edge Case: String containing characters other than '0' or '1'
    for char in binary_string:
        if char != '0' and char != '1':
            raise ValueError("Invalid binary literal: " + binary_string)
    
    # Convert binary to decimal using first principles
    decimal_value = 0
    power = 0
    
    # Process the binary string from right to left
    for i in range(len(binary_string) - 1, -1, -1):
        if binary_string[i] == '1':
            decimal_value += 2 ** power
        power += 1
    
    return decimal_value
# Handled Edge Cases: Input is None, Input is not a string, Empty string, String containing characters other than '0' or '1'

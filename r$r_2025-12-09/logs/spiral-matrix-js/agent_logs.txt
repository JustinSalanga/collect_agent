/**
 * @param {number} size
 * @returns {number[][]}
 */
export const spiralMatrix = (size) => {
  // Edge Case: Negative or zero size input
  if (size <= 0) {
    return [];
  }
  // Edge Case: Size 1 matrix (single element)
  if (size === 1) {
    return [[1]];
  }
  // Initialize matrix with zeros
  const matrix = Array(size).fill().map(() => Array(size).fill(0));
  
  let counter = 1;
  let startRow = 0;
  let endRow = size - 1;
  let startCol = 0;
  let endCol = size - 1;
  
  while (startRow <= endRow && startCol <= endCol) {
    // Fill top row from left to right
    for (let i = startCol; i <= endCol; i++) {
      matrix[startRow][i] = counter++;
    }
    startRow++;
    
    // Fill right column from top to bottom
    for (let i = startRow; i <= endRow; i++) {
      matrix[i][endCol] = counter++;
    }
    endCol--;
    
    // Edge Case: Check if there's still rows to fill (handles even size matrices)
    if (startRow <= endRow) {
      // Fill bottom row from right to left
      for (let i = endCol; i >= startCol; i--) {
        matrix[endRow][i] = counter++;
      }
      endRow--;
    }
    
    // Edge Case: Check if there's still columns to fill (handles even size matrices)
    if (startCol <= endCol) {
      // Fill left column from bottom to top
      for (let i = endRow; i >= startRow; i--) {
        matrix[i][startCol] = counter++;
      }
      startCol++;
    }
  }
  
  return matrix;
};

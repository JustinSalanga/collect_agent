        self.flats_scale = ["A", "Bb", "B", "C", "Db", "D", "Eb", "E", "F", "Gb", "G", "Ab"]
        
        # Determine whether to use sharps or flats based on the tonic
        # Edge Case: Natural notes C and a following special rules for notation
        sharps_keys = {"g", "d", "a", "e", "b", "f#"}
        flats_keys = {"f", "bb", "eb", "ab", "db", "gb"}
        
        if self.tonic in sharps_keys or self.tonic == "c" or self.tonic == "a":
            self.use_sharps = True
        elif self.tonic in flats_keys:
            self.use_sharps = False
        else:
            # Default to sharps for other tonics
            self.use_sharps = True
    def chromatic(self) -> list[str]:
        # Edge Case: Starting from any of the 12 chromatic notes
        if self.use_sharps:
            scale = self.sharps_scale
        else:
            scale = self.flats_scale

        # Edge Case: Handling both sharp and flat notation based on key signature
        tonic_upper = self.tonic[0].upper() + self.tonic[1:] if len(self.tonic) > 1 else self.tonic.upper()
        
        # Handle special case for flat notation
        if not self.use_sharps and len(self.tonic) > 1 and self.tonic[1] == 'b':
            tonic_upper = self.tonic[0].upper() + 'b'

            index = scale.index(tonic_upper)

            # Edge Case: Converting between sharp and flat notation
            equivalent = {
                "A#": "Bb", "Bb": "A#",

                "F#": "Gb", "Gb": "F#",

                index = 0

        result = scale[index:] + scale[:index]

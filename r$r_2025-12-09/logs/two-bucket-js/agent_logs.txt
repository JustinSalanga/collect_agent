export class TwoBucket {
  /**
   * @param {number} size1
   * @param {number} size2
   * @param {number} goal
   * @param {string} start
   */
  constructor(size1, size2, goal, start) {
    // Edge Case: Validate input parameters
    if (size1 <= 0 || size2 <= 0) {
      throw new Error('Bucket sizes must be positive integers.');
    }
    
    if (goal < 0) {
      throw new Error('Goal must be a non-negative integer.');
    }
    
    if (goal > Math.max(size1, size2)) {
      throw new Error('Goal is bigger than the largest bucket.');
    }
    
    // Edge Case: Check if goal is achievable (multiple of GCD)
    const gcd = this.gcd(size1, size2);
    if (goal % gcd !== 0) {
      throw new Error('Goal must be a multiple of the GCD of the sizes of the two buckets.');
    }
    
    this.size1 = size1;
    this.size2 = size2;
    this.goal = goal;
    this.start = start;
  }
  /**
   * Calculate Greatest Common Divisor
   * @param {number} a
   * @param {number} b
   * @return {number}
   */
  gcd(a, b) {
    while (b !== 0) {
      const temp = b;
      b = a % b;
      a = temp;
    }
    return a;
  }
  /**
   * @return {{moves: number, goalBucket: string, otherBucket: number}}
   */
  solve() {
    // Edge Case: Goal is 0
    if (this.goal === 0) {
      return {
        moves: 1,
        goalBucket: this.start === 'one' ? 'two' : 'one',
        otherBucket: this.start === 'one' ? this.size1 : this.size2
      };
    }
    // Edge Case: Goal equals one of the bucket sizes
    if (this.goal === this.size1) {
      if (this.start === 'one') {
        return {
          moves: 1,
          goalBucket: 'one',
          otherBucket: 0
        };
      } else {
        // Need to fill bucket one, which requires 2 moves
        return {
          moves: 2,
          goalBucket: 'one',
          otherBucket: this.size2
        };
      }
    }
    
    if (this.goal === this.size2) {
      if (this.start === 'two') {
        return {
          moves: 1,
          goalBucket: 'two',
          otherBucket: 0
        };
      } else {
        // Need to fill bucket two, which requires 2 moves
        return {
          moves: 2,
          goalBucket: 'two',
          otherBucket: this.size1
        };
      }
    }
    // BFS to find the minimum number of moves
    const visited = new Set();
    const queue = [];
    
    // Initial state based on which bucket to fill first
    if (this.start === 'one') {
      queue.push({ bucket1: this.size1, bucket2: 0, moves: 1 });
      visited.add(`${this.size1},0`);
    } else {
      queue.push({ bucket1: 0, bucket2: this.size2, moves: 1 });
      visited.add(`0,${this.size2}`);
    }
    
    // Edge Case: Starting state is forbidden (empty start bucket and full other bucket)
    const forbiddenState1 = this.start === 'one' ? `0,${this.size2}` : `${this.size1},0`;
    
    // Add iteration limit to prevent infinite loop
    let iterations = 0;
    const maxIterations = 10000;
    
    while (queue.length > 0 && iterations < maxIterations) {
      iterations++;
      const { bucket1, bucket2, moves } = queue.shift();
      
      // Check if we've reached the goal
      if (bucket1 === this.goal) {
        return {
          moves,
          goalBucket: 'one',
          otherBucket: bucket2
        };
      }
      
      if (bucket2 === this.goal) {
        return {
          moves,
          goalBucket: 'two',
          otherBucket: bucket1
        };
      }
      
      // Generate all possible next states
      const nextStates = [];
      
      // 1. Fill bucket1
      nextStates.push({ bucket1: this.size1, bucket2, moves: moves + 1 });
      
      // 2. Fill bucket2
      nextStates.push({ bucket1, bucket2: this.size2, moves: moves + 1 });
      
      // 3. Empty bucket1
      nextStates.push({ bucket1: 0, bucket2, moves: moves + 1 });
      
      // 4. Empty bucket2
      nextStates.push({ bucket1, bucket2: 0, moves: moves + 1 });
      
      // 5. Pour from bucket1 to bucket2
      const pour1To2 = Math.min(bucket1, this.size2 - bucket2);
      nextStates.push({ 
        bucket1: bucket1 - pour1To2, 
        bucket2: bucket2 + pour1To2, 
        moves: moves + 1 
      });
      
      // 6. Pour from bucket2 to bucket1
      const pour2To1 = Math.min(bucket2, this.size1 - bucket1);
      nextStates.push({ 
        bucket1: bucket1 + pour2To1, 
        bucket2: bucket2 - pour2To1, 
        moves: moves + 1 
      });
      
      // Add valid next states to queue
      for (const state of nextStates) {
        const stateKey = `${state.bucket1},${state.bucket2}`;
        
        // Edge Case: Skip forbidden state (starting bucket empty, other full)
        if (stateKey === forbiddenState1) continue;
        
        // Edge Case: Skip already visited states
        if (visited.has(stateKey)) continue;
        
        // Edge Case: Skip states that exceed bucket capacities
        if (state.bucket1 > this.size1 || state.bucket2 > this.size2) continue;
        if (state.bucket1 < 0 || state.bucket2 < 0) continue;
        
        visited.add(stateKey);
        queue.push(state);
      }
    }
    
    // Edge Case: Goal is not achievable (should not happen given GCD check)
    throw new Error('Goal is not achievable.');
  }

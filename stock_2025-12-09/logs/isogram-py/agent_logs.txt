def is_isogram(string: str) -> bool:
    # Edge Case: Empty string is considered an isogram
    if not string:
        return True
    
    # Normalize the string: convert to lowercase and filter only alphabetic characters
    normalized_chars = [char.lower() for char in string if char.isalpha()]
    
    # Edge Case: String with no alphabetic characters (only spaces, hyphens, etc.) is considered an isogram
    if not normalized_chars:
        return True
    
    # Check for duplicate letters by comparing the length of the list of characters
    # with the length of the set of those characters
    return len(normalized_chars) == len(set(normalized_chars))

export const solve = (puzzle) => {
  // Edge Case: Empty or invalid puzzle format
  if (!puzzle || typeof puzzle !== 'string') {
    return null;
  }
  // Parse the puzzle
  const parts = puzzle.replace(/\s/g, '').split(/[\+\=]/);
  
  // Edge Case: Invalid puzzle format (not enough parts)
  if (parts.length < 3) {
    return null;
  }
  
  const addends = parts.slice(0, -1);
  const result = parts[parts.length - 1];
  const allWords = [...addends, result];
  
  // Edge Case: Empty words in puzzle
  if (allWords.some(word => !word)) {
    return null;
  }
  
  // Get all unique letters
  const letters = [...new Set(allWords.join('').split(''))];
  
  // Edge Case: More than 10 unique letters (impossible to solve)
  if (letters.length > 10) {
    return null;
  }
  
  // Get leading letters (first letter of each word)
  const leadingLetters = [...new Set(allWords.map(word => word[0]))];
  
  // Function to convert word to number based on mapping
  const wordToNumber = (word, mapping) => {
    return parseInt(word.split('').map(char => mapping[char]).join(''), 10);
  };

  const isValidMapping = (mapping) => {
    // Edge Case: Leading letters cannot be zero
    if (leadingLetters.some(letter => mapping[letter] === 0)) {

    const addendValues = addends.map(word => wordToNumber(word, mapping));
    const resultValue = wordToNumber(result, mapping);
    
    // Edge Case: Check if sum is correct
    return addendValues.reduce((sum, val) => sum + val, 0) === resultValue;
  };

  const generatePermutations = (digits, length) => {

      const rest = digits.slice(0, i).concat(digits.slice(i + 1));
      const restPermutations = generatePermutations(rest, length - 1);
      for (const perm of restPermutations) {
        permutations.push([digits[i], ...perm]);
      }
    }
    return permutations;
  };

export const spiralMatrix = (size) => {
  // Edge Case: Invalid input (zero or negative size)
  if (size <= 0) {
    return [];
  }
  // Edge Case: Size 1 matrix (single element)
  if (size === 1) {
    return [[1]];
  }
  // Initialize matrix with zeros
  const matrix = Array(size).fill().map(() => Array(size).fill(0));
  
  let counter = 1;
  let startRow = 0;
  let endRow = size - 1;
  let startCol = 0;
  let endCol = size - 1;
  
  while (startRow <= endRow && startCol <= endCol) {
    // Top row: left to right
    for (let i = startCol; i <= endCol; i++) {
      matrix[startRow][i] = counter++;
    }
    startRow++;
    
    // Right column: top to bottom
    for (let i = startRow; i <= endRow; i++) {
      matrix[i][endCol] = counter++;
    }
    endCol--;
    
    // Bottom row: right to left (if there's still a row to fill)
    if (startRow <= endRow) {
      for (let i = endCol; i >= startCol; i--) {
        matrix[endRow][i] = counter++;
      }
      endRow--;
    }
    
    // Left column: bottom to top (if there's still a column to fill)
    if (startCol <= endCol) {
      for (let i = endRow; i >= startRow; i--) {
        matrix[i][startCol] = counter++;
      }
      startCol++;
    }
  }
  
  return matrix;
};

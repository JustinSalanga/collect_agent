def primes(limit: int) -> list[int]:
    # Edge Case: Handle negative numbers and numbers less than 2
    if limit < 2:
        return []
    
    # Edge Case: Handle limit of 2
    if limit == 2:
        return [2]
    
    # Initialize a boolean array to track prime status
    # Index i represents number i, True means potentially prime
    is_prime = [True] * (limit + 1)
    is_prime[0] = False  # 0 is not prime
    is_prime[1] = False  # 1 is not prime
    
    # Implement Sieve of Eratosthenes
    for i in range(2, int(limit**0.5) + 1):
        if is_prime[i]:
            # Mark all multiples of i as not prime
            # Start from i*i as all smaller multiples have been marked already
            for j in range(i * i, limit + 1, i):
                is_prime[j] = False
    
    # Collect all prime numbers
    result = []
    for i in range(2, limit + 1):
        if is_prime[i]:
            result.append(i)
    
    return result

/**
 * @param {string} planet
 * @param {number} seconds
 * @returns {number}
 */
export const age = (planet, seconds) => {
  // Edge Case: Invalid planet name input
  if (typeof planet !== 'string' || planet === '') {
    throw new Error('not a planet');
  }
  // Edge Case: Negative or zero age in seconds
  if (typeof seconds !== 'number' || seconds <= 0) {
    throw new Error('seconds must be a positive number');
  }
  // Edge Case: Case sensitivity in planet name validation
  const normalizedPlanet = planet.charAt(0).toUpperCase() + planet.slice(1).toLowerCase();
  
  const planetOrbitalPeriods = {
    'Mercury': 0.2408467,
    'Venus': 0.61519726,
    'Earth': 1.0,
    'Mars': 1.8808158,
    'Jupiter': 11.862615,
    'Saturn': 29.447498,
    'Uranus': 84.016846,
    'Neptune': 164.79132
  };
  // Edge Case: Invalid planet name input (not in our solar system)
  if (!planetOrbitalPeriods[normalizedPlanet]) {
    throw new Error('not a planet');
  }
  // Earth year in seconds
  const earthYearInSeconds = 31557600;
  
  // Edge Case: Extremely large age values causing precision issues
  // Calculate age in Earth years first, then convert to the target planet
  const earthAge = seconds / earthYearInSeconds;
  const planetAge = earthAge / planetOrbitalPeriods[normalizedPlanet];
  
  // Round to two decimal places
  return Math.round(planetAge * 100) / 100;
};
// Handled Edge Cases: Invalid planet name input, Negative or zero age in seconds, Case sensitivity in planet name validation, Extremely large age values causing precision issues

def is_isogram(string: str) -> bool:
    # Edge Case: Empty string is considered an isogram
    if not string:
        return True
    
    # Convert to lowercase for case-insensitive comparison
    lowercase_string = string.lower()
    
    # Track seen letters
    seen_letters = set()
    
    # Iterate through each character in the string
    for char in lowercase_string:
        # Edge Case: Ignore non-alphabetic characters (spaces, hyphens, etc.)
        if not char.isalpha():
            continue
            
        # Edge Case: If letter has been seen before, it's not an isogram
        if char in seen_letters:
            return False
            
        seen_letters.add(char)
    
    return True

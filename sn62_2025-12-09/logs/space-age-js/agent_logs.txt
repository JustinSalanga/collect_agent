/**
 * @param {string} planet
 * @param {number} seconds
 * @returns {number}
 */
export const age = (planet, seconds) => {
  // Edge Case: Invalid input - negative seconds
  if (seconds < 0) {
    throw new Error('seconds must be non-negative');
  }
  // Edge Case: Invalid input - non-numeric seconds
  if (typeof seconds !== 'number' || isNaN(seconds)) {
    throw new Error('seconds must be a number');
  }
  const earthYearInSeconds = 31557600;
  
  const planetOrbitalPeriods = {
    'mercury': 0.2408467,
    'venus': 0.61519726,
    'earth': 1.0,
    'mars': 1.8808158,
    'jupiter': 11.862615,
    'saturn': 29.447498,
    'uranus': 84.016846,
    'neptune': 164.79132
  };
  // Edge Case: Invalid planet name (not a string)
  if (typeof planet !== 'string') {
    throw new Error('not a planet');
  }
  // Edge Case: Invalid planet name (case insensitive)
  const normalizedPlanet = planet.toLowerCase();
  if (!planetOrbitalPeriods.hasOwnProperty(normalizedPlanet)) {
    throw new Error('not a planet');
  }
  const orbitalPeriod = planetOrbitalPeriods[normalizedPlanet];
  const ageInEarthYears = seconds / earthYearInSeconds;
  const ageInPlanetYears = ageInEarthYears / orbitalPeriod;
  
  // Round to two decimal places
  return Math.round(ageInPlanetYears * 100) / 100;
};
// Handled Edge Cases: negative seconds, non-numeric seconds, invalid planet name (case insensitive), invalid planet name (not a string)

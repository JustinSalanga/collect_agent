/**
 * @param {number[][]} matrix
 * @returns {{row: number, column: number}[]}
 */
export const saddlePoints = (matrix) => {
  // Edge Case: Empty matrix or matrix with no columns
  if (!matrix || matrix.length === 0 || matrix[0].length === 0) {
    return [];
  }
  const rows = matrix.length;
  const cols = matrix[0].length;
  const result = [];
  // Precompute row maxima
  const rowMax = new Array(rows).fill(Number.NEGATIVE_INFINITY);
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      if (matrix[i][j] > rowMax[i]) {
        rowMax[i] = matrix[i][j];
      }
    }
  }
  // Precompute column minima
  const colMin = new Array(cols).fill(Number.POSITIVE_INFINITY);
  for (let j = 0; j < cols; j++) {
    for (let i = 0; i < rows; i++) {
      if (matrix[i][j] < colMin[j]) {
        colMin[j] = matrix[i][j];
      }
    }
  }
  // Find saddle points
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      // Edge Case: Value is both maximum in its row and minimum in its column
      if (matrix[i][j] === rowMax[i] && matrix[i][j] === colMin[j]) {
        result.push({ row: i + 1, column: j + 1 }); // 1-indexed
      }
    }
  }
  return result;
};

// This is only a SKELETON file for the 'Saddle Points' exercise. It's been provided as a
// convenience to get you started writing code faster.
//
/**
 * @param {number[][]} matrix
 * @returns {{row: number, column: number}[]}
 */
export const saddlePoints = (matrix) => {
  // Edge Case: Empty matrix
  if (!matrix || matrix.length === 0) {
    return [];
  }
  // Edge Case: Matrix with empty rows
  const numRows = matrix.length;
  const numCols = matrix[0] ? matrix[0].length : 0;
  
  if (numCols === 0) {
    return [];
  }
  // Precompute row maximums
  const rowMaxes = [];
  for (let i = 0; i < numRows; i++) {
    let max = matrix[i][0];
    for (let j = 1; j < numCols; j++) {
      if (matrix[i][j] > max) {
        max = matrix[i][j];
      }
    }
    rowMaxes[i] = max;
  }
  // Precompute column minimums
  const colMins = [];
  for (let j = 0; j < numCols; j++) {
    let min = matrix[0][j];
    for (let i = 1; i < numRows; i++) {
      if (matrix[i][j] < min) {
        min = matrix[i][j];
      }
    }
    colMins[j] = min;
  }
  // Find saddle points
  const result = [];
  for (let i = 0; i < numRows; i++) {
    for (let j = 0; j < numCols; j++) {
      const value = matrix[i][j];
      // Edge Case: Value is both maximum in its row and minimum in its column
      if (value === rowMaxes[i] && value === colMins[j]) {
        result.push({ row: i + 1, column: j + 1 }); // 1-indexed
      }
    }
  }
  return result;
};

// This is only a SKELETON file for the 'Saddle Points' exercise. It's been provided as a

/**
 * @param {number[][]} matrix
 * @returns {{row: number, column: number}[]}
 */
export const saddlePoints = (matrix) => {
  // Edge Case: Empty matrix or matrix with no columns
  if (!matrix || matrix.length === 0 || matrix[0].length === 0) {
    return [];
  }
  const rows = matrix.length;
  const cols = matrix[0].length;
  const result = [];
  // Precompute row maxima
  const rowMax = new Array(rows);
  for (let i = 0; i < rows; i++) {
    rowMax[i] = Math.max(...matrix[i]);
  }
  // Precompute column minima
  const colMin = new Array(cols);
  for (let j = 0; j < cols; j++) {
    const column = matrix.map(row => row[j]);
    colMin[j] = Math.min(...column);
  }
  // Find saddle points
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      const value = matrix[i][j];
      // Edge Case: Value is both the maximum in its row and minimum in its column
      if (value === rowMax[i] && value === colMin[j]) {
        result.push({ row: i + 1, column: j + 1 }); // 1-indexed
      }
    }
  }
  return result;
};
// Handled Edge Cases: Empty matrix, matrix with no columns, value being maximum in row and minimum in column

// This is only a SKELETON file for the 'Saddle Points' exercise. It's been provided as a

/**
 * @param {number[][]} matrix
 * @returns {{row: number, column: number}[]}
 */
export const saddlePoints = (matrix) => {
  // Edge Case: Empty matrix or matrix with no columns
  if (!matrix || matrix.length === 0 || matrix[0].length === 0) {
    return [];
  }
  const rows = matrix.length;
  const cols = matrix[0].length;
  const saddlePoints = [];
  // Precompute row maxima
  const rowMax = new Array(rows).fill(Number.NEGATIVE_INFINITY);
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      if (matrix[i][j] > rowMax[i]) {
        rowMax[i] = matrix[i][j];
      }
    }
  }
  // Precompute column minima
  const colMin = new Array(cols).fill(Number.POSITIVE_INFINITY);
  for (let j = 0; j < cols; j++) {
    for (let i = 0; i < rows; i++) {
      if (matrix[i][j] < colMin[j]) {
        colMin[j] = matrix[i][j];
      }
    }
  }
  // Find saddle points
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      // Edge Case: Value is both maximum in its row and minimum in its column
      if (matrix[i][j] === rowMax[i] && matrix[i][j] === colMin[j]) {
        saddlePoints.push({ row: i + 1, column: j + 1 }); // 1-indexed
      }
    }
  }
  return saddlePoints;
};

 * @returns {{row: number, column: number}[]}

  // Edge Case: Empty matrix or matrix with no columns

    }
  }
  
  return result;

// This is only a SKELETON file for the 'Saddle Points' exercise. It's been provided as a
// convenience to get you started writing code faster.

/**
 * @param {number[][]} matrix
 * @returns {{row: number, column: number}[]}
 */
export const saddlePoints = (matrix) => {
  // Edge Case: Empty matrix or matrix with no columns
  if (!matrix || matrix.length === 0 || matrix[0].length === 0) {
    return [];
  }
  const rows = matrix.length;
  const cols = matrix[0].length;
  
  // Precompute row maxima
  const rowMax = new Array(rows);
  for (let i = 0; i < rows; i++) {
    rowMax[i] = Math.max(...matrix[i]);
  }
  
  // Precompute column minima
  const colMin = new Array(cols);
  for (let j = 0; j < cols; j++) {
    const column = matrix.map(row => row[j]);
    colMin[j] = Math.min(...column);
  }
  
  const result = [];
  
  // Check each cell to see if it's a saddle point
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      const value = matrix[i][j];
      // A saddle point is maximum in its row and minimum in its column
      if (value === rowMax[i] && value === colMin[j]) {
        // Convert to 1-indexed coordinates
        result.push({ row: i + 1, column: j + 1 });
      }
    }
  }
  
  return result;
};
// Handled Edge Cases: Empty matrix, matrix with no columns

/**
 * @param {number[][]} matrix
 * @returns {{row: number, column: number}[]}
 */
export const saddlePoints = (matrix) => {
  // Edge Case: Empty matrix or matrix with no columns
  if (!matrix || matrix.length === 0 || matrix[0].length === 0) {
    return [];
  }
  const rows = matrix.length;
  const cols = matrix[0].length;
  const saddlePoints = [];
  // Precompute row maxima
  const rowMax = new Array(rows);
  for (let i = 0; i < rows; i++) {
    rowMax[i] = Math.max(...matrix[i]);
  }
  // Precompute column minima
  const colMin = new Array(cols);
  for (let j = 0; j < cols; j++) {
    const column = matrix.map(row => row[j]);
    colMin[j] = Math.min(...column);
  }
  // Find saddle points
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      const value = matrix[i][j];
      // Edge Case: Value is both maximum in its row and minimum in its column
      if (value === rowMax[i] && value === colMin[j]) {
        saddlePoints.push({ row: i + 1, column: j + 1 }); // 1-indexed
      }
    }
  }
  return saddlePoints;
  // Handled Edge Cases: Empty matrix, matrix with no columns, value is both maximum in its row and minimum in its column
};

// This is only a SKELETON file for the 'Saddle Points' exercise. It's been provided as a
// convenience to get you started writing code faster.
//
/**
 * @param {number[][]} matrix
 * @returns {{row: number, column: number}[]}
 */
export const saddlePoints = (matrix) => {
  // Edge Case: Empty matrix or matrix with no rows
  if (!matrix || matrix.length === 0) {
    return [];
  }
  const numRows = matrix.length;
  // Edge Case: Matrix with no columns
  if (matrix[0].length === 0) {
    return [];
  }
  const numCols = matrix[0].length;
  // Precompute row maximums
  const rowMax = new Array(numRows);
  for (let i = 0; i < numRows; i++) {
    // Edge Case: Row with no elements
    if (matrix[i].length === 0) {
      rowMax[i] = -Infinity;
    } else {
      rowMax[i] = Math.max(...matrix[i]);
    }
  }
  // Precompute column minimums
  const colMin = new Array(numCols);
  for (let j = 0; j < numCols; j++) {
    const column = [];
    for (let i = 0; i < numRows; i++) {
      // Edge Case: Inconsistent row lengths
      if (j < matrix[i].length) {
        column.push(matrix[i][j]);
      }
    }
    // Edge Case: Column with no elements
    if (column.length === 0) {
      colMin[j] = Infinity;
    } else {
      colMin[j] = Math.min(...column);
    }
  }
  const result = [];
  for (let i = 0; i < numRows; i++) {
    for (let j = 0; j < matrix[i].length; j++) {
      const value = matrix[i][j];
      // Check if the value is the largest in its row and smallest in its column
      if (value === rowMax[i] && value === colMin[j]) {
        // Convert to 1-indexed coordinates
        result.push({ row: i + 1, column: j + 1 });
      }
    }
  }
  return result;
  // Handled Edge Cases: Empty matrix, matrix with no rows, matrix with no columns, row with no elements, inconsistent row lengths, column with no elements
};

    }
  }

/**
 * @param {number[][]} matrix
 * @returns {{row: number, column: number}[]}
 */
export const saddlePoints = (matrix) => {
  // Edge Case: Empty matrix or matrix with no columns
  if (!matrix || matrix.length === 0 || matrix[0].length === 0) {
    return [];
  }
  const rows = matrix.length;
  const cols = matrix[0].length;
  const result = [];
  // Precompute row maxima
  const rowMax = new Array(rows).fill(Number.NEGATIVE_INFINITY);
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      if (matrix[i][j] > rowMax[i]) {
        rowMax[i] = matrix[i][j];
      }
    }
  }
  // Precompute column minima
  const colMin = new Array(cols).fill(Number.POSITIVE_INFINITY);
  for (let j = 0; j < cols; j++) {
    for (let i = 0; i < rows; i++) {
      if (matrix[i][j] < colMin[j]) {
        colMin[j] = matrix[i][j];
      }
    }
  }
  // Find saddle points
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      // Edge Case: Value is both maximum in its row and minimum in its column
      if (matrix[i][j] === rowMax[i] && matrix[i][j] === colMin[j]) {
        result.push({ row: i + 1, column: j + 1 }); // 1-indexed
      }
    }
  }
  // Handled Edge Cases: Empty matrix, matrix with no columns, values that are both row max and column min
  return result;
};

    """
    Calculate the square of the sum of the first N natural numbers.
    
    Formula: (n(n+1)/2)^2
    
    Args:
        number: A positive integer representing N
        
    Returns:
        The square of the sum of first N natural numbers
        
    Raises:
        ValueError: If number is negative
    """
    # Edge Case: Negative number
    if number < 0:
        raise ValueError("Number must be non-negative")
    
    # Edge Case: Zero
    if number == 0:
        return 0
    
    total = number * (number + 1) // 2
    return total * total
    # Handled Edge Cases: Negative number, Zero

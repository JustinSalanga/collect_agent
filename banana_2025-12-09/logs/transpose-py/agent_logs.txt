def transpose(text: str) -> str:
    """
    Transpose the input text by converting rows to columns and columns to rows.
    
    If rows have different lengths, shorter rows are padded with spaces on the right
    to match the length of the longest row.
    
    Args:
        text: A string containing the input text with rows separated by newlines
        
    Returns:
        A string representing the transposed text
        
    Examples:
        >>> transpose("ABC\
DEF")
        'AD\
BE\
CF'
        >>> transpose("ABC\
DE")
        'AD\
BE\
C'
        >>> transpose("AB\
DEF")
        'AD\
BE\
 F'
    """
    # Edge Case: Empty input string
    if not text:
        return ""
    
    # Split the input text into rows
    rows = text.split('
')
    
    # Edge Case: Single row input
    if len(rows) == 1:
        return '
'.join(rows[0])
    
    # Find the maximum length among all rows
    max_length = max(len(row) for row in rows) if rows else 0
    
    # Edge Case: All rows are empty
    if max_length == 0:
        return ""
    
    # Pad each row to the maximum length with spaces
    padded_rows = [row.ljust(max_length) for row in rows]
    
    # Transpose the matrix
    transposed = []
    for col_index in range(max_length):
        column = ''.join(row[col_index] for row in padded_rows)
        transposed.append(column)
    
    # Remove trailing spaces from each row of the transposed matrix
    result_rows = [row.rstrip(' ') for row in transposed]
    
    return '
'.join(result_rows)

def transpose(text: str) -> str:
    """
    Transpose the input text matrix.
    
    Args:
        text: A string representing a matrix of characters separated by newlines
        
    Returns:
        A string representing the transposed matrix
        
    The transpose operation converts rows to columns and columns to rows.
    If rows have different lengths, they are padded with spaces on the left
    (but not on the right) to make all rows the same length before transposing.
    """
    if not text:
        return ""
    
    # Split the input into lines
    lines = text.split('
')
    
    # Edge Case: Single line input
    if len(lines) == 1:
        if not lines[0]:
            return ""
        return '
'.join(lines[0])
    
    # Edge Case: Empty lines only
    if all(not line for line in lines):
        return ""
    
    # Find the maximum length among all lines
    max_length = max(len(line) for line in lines) if lines else 0
    
    # Edge Case: All lines are empty
    if max_length == 0:
        return ""
    
    # Pad each line to the left to make all lines the same length
    padded_lines = []
    for line in lines:
        padding_needed = max_length - len(line)
        padded_line = ' ' * padding_needed + line
        padded_lines.append(padded_line)
    
    # Transpose the matrix
    transposed = []
    for col in range(max_length):
        new_row = ''.join(padded_lines[row][col] for row in range(len(padded_lines)))
        transposed.append(new_row)
    
    # Remove trailing spaces from each row but preserve internal structure
    result_lines = []
    for line in transposed:
        # Edge Case: Line with only spaces
        if line.isspace() or not line:
            result_lines.append(line.rstrip())
        else:
            result_lines.append(line.rstrip())
    
    return '
'.join(result_lines)

def transpose(text: str) -> str:
    """
    Transpose the input text matrix.
    
    Args:
        text (str): Input text with rows separated by newlines
        
    Returns:
        str: Transposed text with rows separated by newlines
        
    The transpose operation converts rows to columns and columns to rows.
    For inputs with rows of different lengths:
    - Pad to the left with spaces (don't pad to the right)
    - Preserve all characters from the input
    """
    # Edge Case: Empty input string
    if not text:
        return ""
    
    # Split the input into lines
    lines = text.split('
')
    
    # Edge Case: Single line input
    if len(lines) == 1:
        if not lines[0]:
            return ""
        return '
'.join(lines[0])
    
    # Edge Case: Empty lines only
    if all(not line for line in lines):
        return ""
    
    # Find the maximum length among all lines
    max_length = max(len(line) for line in lines) if lines else 0
    
    # Edge Case: All lines are empty
    if max_length == 0:
        return ""
    
    # Pad each line to the left to match the maximum length
    padded_lines = []
    for line in lines:
        padding_needed = max_length - len(line)
        padded_line = ' ' * padding_needed + line
        padded_lines.append(padded_line)
    
    # Transpose the matrix
    transposed = []
    for col in range(max_length):
        new_row = ''.join(padded_lines[row][col] for row in range(len(padded_lines)))
        transposed.append(new_row)
    
    # Remove trailing spaces from each row but preserve internal structure
    result_lines = []
    for line in transposed:
        # Edge Case: Line with only spaces
        if line.isspace() or not line:
            result_lines.append(line.rstrip())
        else:
            result_lines.append(line.rstrip())
    
    return '
'.join(result_lines)

def transpose(text: str) -> str:
    """
    Transpose the input text by converting rows to columns and columns to rows.
    
    If rows have different lengths, shorter rows are padded with spaces on the right
    to match the length of the longest row.
    
    Args:
        text: A string containing the input text with rows separated by newlines
        
    Returns:
        A string representing the transposed text
        
    Examples:
        >>> transpose("ABC\
DEF")
        'AD\
BE\
CF'
        >>> transpose("ABC\
DE")
        'AD\
BE\
C'
    """
    # Edge Case: Empty input string
    if not text:
        return ""
    
    # Split the input text into rows
    rows = text.split('
')
    
    # Edge Case: Single row input
    if len(rows) == 1:
        return '
'.join(rows[0])
    
    # Find the maximum length among all rows
    max_length = max(len(row) for row in rows) if rows else 0
    
    # Edge Case: All rows are empty
    if max_length == 0:
        return ""
    
    # Pad each row to the maximum length with spaces
    padded_rows = [row.ljust(max_length) for row in rows]
    
    # Transpose the matrix
    transposed = []
    for col_index in range(max_length):
        column = ''.join(row[col_index] for row in padded_rows)
        # Remove trailing spaces from the column
        transposed.append(column.rstrip())
    
    # Edge Case: Resulting empty rows after stripping
    # This is handled by the rstrip() above which may produce empty strings
    
    return '
'.join(transposed)

def transpose(text: str) -> str:
    """
    Transpose the input text matrix.
    
    Args:
        text: A string representing a matrix of characters separated by newlines
        
    Returns:
        A string representing the transposed matrix
        
    Example:
        >>> transpose("ABC\
DEF")
        'AD\
BE\
CF'
    """
    # Edge Case: empty input string
    if not text:
        return ""
    
    # Split the input into lines
    lines = text.split('
')
    
    # Edge Case: single line input
    if len(lines) == 1:
        return '
'.join(lines[0]) if lines[0] else ""
    
    # Edge Case: empty lines
    if all(not line for line in lines):
        return ""
    
    # Find the maximum length among all lines
    max_length = max(len(line) for line in lines) if lines else 0
    
    # Edge Case: all lines are empty
    if max_length == 0:
        return ""
    
    # Pad each line to the maximum length with spaces on the right
    padded_lines = [line.ljust(max_length) for line in lines]
    
    # Transpose the matrix
    transposed = []
    for col in range(max_length):
        # Build each row of the transposed matrix
        new_row = ''.join(padded_lines[row][col] for row in range(len(padded_lines)))
        transposed.append(new_row)
    
    # Remove trailing spaces from each row but preserve leading spaces
    result_lines = []
    for line in transposed:
        # Edge Case: line with only spaces
        if line.isspace() or not line:
            result_lines.append(line)
        else:
            # Remove trailing spaces but keep leading spaces
            result_lines.append(line.rstrip())
    
    return '
'.join(result_lines)

def transpose(text: str) -> str:
    """
    Transpose the input text matrix.
    
    Args:
        text (str): Input text with rows separated by newlines
        
    Returns:
        str: Transposed text with rows separated by newlines
        
    The transpose operation converts rows to columns and columns to rows.
    For inputs with rows of different lengths:
    - Pad to the left with spaces
    - Don't pad to the right
    """
    # Edge Case: Empty input string
    if not text:
        return ""
    
    # Edge Case: Input with only newlines
    lines = text.split('
')
    if all(line == '' for line in lines):
        return '
'.join(['' for _ in lines[:-1]]) if lines else ""
    
    # Edge Case: Single row input
    if len(lines) == 1:
        return '
'.join(list(lines[0])) if lines[0] else ""
    
    # Find the maximum length among all rows
    max_length = max(len(line) for line in lines) if lines else 0
    
    # Edge Case: All rows are empty
    if max_length == 0:
        return '
'.join(['' for _ in lines[:-1]]) if len(lines) > 1 else ""
    
    # Pad each row to the left to match the maximum length
    padded_lines = []
    for line in lines:
        padding_needed = max_length - len(line)
        padded_line = ' ' * padding_needed + line
        padded_lines.append(padded_line)
    
    # Transpose the matrix
    transposed_rows = []
    for col_index in range(max_length):
        transposed_row = ''.join(padded_lines[row_index][col_index] 
                                for row_index in range(len(padded_lines)))
        transposed_rows.append(transposed_row)
    
    # Remove trailing spaces from each row but preserve internal structure
    result_rows = []
    for row in transposed_rows:
        # Edge Case: Rows that might be all spaces
        if all(c == ' ' for c in row):
            result_rows.append("")
        else:
            result_rows.append(row.rstrip(' '))
    
    # Edge Case: Remove trailing empty rows that might result from rstrip
    # Actually, we should not remove empty rows as they might be meaningful in transpose
    # Let's just rstrip each row but keep all rows
    result_rows = [row.rstrip(' ') for row in transposed_rows]
    
    return '
'.join(result_rows)

def transpose(text: str) -> str:
    """
    Transpose the input text by converting rows to columns and columns to rows.
    
    If rows have different lengths, shorter rows are padded with spaces on the right
    to match the length of the longest row.
    
    Args:
        text: A string containing the input text with rows separated by newlines
        
    Returns:
        A string representing the transposed text
        
    Examples:
        >>> transpose("ABC\
DEF")
        'AD\
BE\
CF'
        >>> transpose("ABC\
DE")
        'AD\
BE\
C'
        >>> transpose("AB\
DEF")
        'AD\
BE\
 F'
    """
    # Edge Case: Empty input string
    if not text:
        return ""
    
    # Edge Case: Input with no newlines (single row)
    if '
' not in text:
        return '
'.join(list(text))
    
    lines = text.split('
')
    
    # Edge Case: Input with empty lines
    if not lines or all(line == "" for line in lines):
        if len(lines) <= 1:
            return ""
        else:
            return '
'.join([''] * len(lines))
    
    # Find the maximum length among all rows
    max_length = max(len(line) for line in lines) if lines else 0
    
    # Edge Case: All rows are empty
    if max_length == 0:
        return '
'.join([''] * len(lines))
    
    # Pad each line to the maximum length with spaces
    padded_lines = [line.ljust(max_length) for line in lines]
    
    # Transpose the matrix
    transposed = []
    for col in range(max_length):
        new_row = ''.join(padded_lines[row][col] for row in range(len(padded_lines)))
        transposed.append(new_row)
    
    # Remove trailing spaces from each row but preserve leading spaces
    result_lines = []
    for line in transposed:
        # Edge Case: Lines with only spaces
        if line.isspace() or not line:
            result_lines.append(line)
        else:
            # Remove trailing spaces but keep leading spaces
            result_lines.append(line.rstrip())
    
    return '
'.join(result_lines)

def transpose(text: str) -> str:
    """
    Transpose the input text matrix.
    
    Args:
        text: A string representing a matrix of characters separated by newlines
        
    Returns:
        A string representing the transposed matrix
        
    Example:
        >>> transpose("ABC\
DEF")
        'AD\
BE\
CF'
    """
    # Edge Case: empty input string
    if not text:
        return ""
    
    # Split the input into lines
    lines = text.split('
')
    
    # Edge Case: single line input
    if len(lines) == 1:
        return '
'.join(lines[0]) if lines[0] else ""
    
    # Edge Case: empty lines
    if all(not line for line in lines):
        return ""
    
    # Find the maximum length among all lines
    max_length = max(len(line) for line in lines) if lines else 0
    
    # Edge Case: all lines are empty
    if max_length == 0:
        return ""
    
    # Pad each line to the maximum length with spaces on the right
    padded_lines = [line.ljust(max_length) for line in lines]
    
    # Transpose the matrix
    transposed = []
    for col in range(max_length):
        column = ''.join(padded_lines[row][col] for row in range(len(padded_lines)))
        transposed.append(column)
    
    # Remove trailing spaces from each row of the transposed matrix
    result_lines = [line.rstrip() for line in transposed]
    
    return '
'.join(result_lines)

def transpose(text: str) -> str:
    """
    Transpose the input text by converting rows to columns and columns to rows.
    
    If rows have different lengths, shorter rows are padded with spaces on the right
    to match the length of the longest row.
    
    Args:
        text: A string containing the input text with rows separated by newlines
        
    Returns:
        A string representing the transposed text
        
    Examples:
        >>> transpose("ABC\
DEF")
        'AD\
BE\
CF'
        >>> transpose("ABC\
DE")
        'AD\
BE\
C'
        >>> transpose("AB\
DEF")
        'AD\
BE\
 F'
    """
    # Edge Case: Empty input string
    if not text:
        return ""
    
    # Split the input text into rows
    rows = text.split('
')
    
    # Edge Case: Single row input
    if len(rows) == 1:
        return '
'.join(rows[0])
    
    # Find the maximum length among all rows
    max_length = max(len(row) for row in rows) if rows else 0
    
    # Edge Case: All rows are empty
    if max_length == 0:
        return ""
    
    # Pad each row to the maximum length with spaces
    padded_rows = [row.ljust(max_length) for row in rows]
    
    # Transpose the matrix
    transposed = []
    for col_index in range(max_length):
        column = ''.join(row[col_index] for row in padded_rows)
        transposed.append(column)
    
    # Remove trailing spaces from each row of the transposed matrix
    # but preserve leading and internal spaces
    result_rows = []
    for row in transposed:
        # Edge Case: Row with only spaces
        if row.strip() == '':
            result_rows.append(row.rstrip())
        else:
            # Edge Case: Row with trailing spaces that need to be preserved
            result_rows.append(row.rstrip())
    
    return '
'.join(result_rows)

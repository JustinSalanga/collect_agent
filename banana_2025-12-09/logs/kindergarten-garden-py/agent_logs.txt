class Garden:
    """
    A class representing a garden with plants assigned to students.
    
    The garden consists of two rows of plants, and each student is responsible
    for four plants (two from each row). Students are assigned plants based
    on their position in an alphabetically sorted list.
    """
    
    PLANT_MAP = {
        'G': 'Grass',
        'C': 'Clover',
        'R': 'Radishes',
        'V': 'Violets'
    }
    
    DEFAULT_STUDENTS = [
        "Alice", "Bob", "Charlie", "David", "Eve", "Fred",
        "Ginny", "Harriet", "Ileana", "Joseph", "Kincaid", "Larry"
    ]
    
    def __init__(self, diagram: str, students: list[str] | None = None) -> None:
        """
        Initialize a Garden with a diagram and optional student list.
        
        Args:
            diagram: A string representing the garden layout with two rows
            students: An optional list of student names to override the default roster
        """
        # Edge Case: Handle empty diagram
        if not diagram:
            self.rows = ["", ""]
        else:
            # Split diagram into rows
            rows = diagram.split('
')
            # Edge Case: Handle diagram with only one row
            if len(rows) == 1:
                self.rows = [rows[0], ""]
            # Edge Case: Handle diagram with more than two rows (take only first two)
            elif len(rows) >= 2:
                self.rows = [rows[0], rows[1]]
        
        # Set up student roster
        # Edge Case: Handle None students parameter (use default)
        if students is None:
            self.students = sorted(self.DEFAULT_STUDENTS)
        else:
            # Edge Case: Handle empty students list
            self.students = sorted(students) if students else []
        
        # Edge Case: Handle mismatched row lengths by taking minimum length
        if self.rows[0] and self.rows[1]:
            min_length = min(len(self.rows[0]), len(self.rows[1]))
            self.rows = [self.rows[0][:min_length], self.rows[1][:min_length]]
    
    def plants(self, student: str) -> list[str]:
        """
        Get the list of plants belonging to a specific student.
        
        Args:
            student: The name of the student
            
        Returns:
            A list of four plant names belonging to the student
            
        Raises:
            ValueError: If the student is not in the roster
        """
        # Edge Case: Handle student not in roster
        if student not in self.students:
            raise ValueError(f"Student {student} not found in roster")
        
        # Edge Case: Handle empty student roster
        if not self.students:
            return []
        
        # Find student's position
        position = self.students.index(student)
        
        # Each student gets 2 consecutive plants from each row
        start_index = position * 2
        
        result = []
        
        # Get plants from first row
        # Edge Case: Handle index out of bounds for first row
        if start_index < len(self.rows[0]):
            result.append(self.PLANT_MAP.get(self.rows[0][start_index], ""))
        if start_index + 1 < len(self.rows[0]):
            result.append(self.PLANT_MAP.get(self.rows[0][start_index + 1], ""))
            
        # Get plants from second row
        # Edge Case: Handle index out of bounds for second row
        if start_index < len(self.rows[1]):
            result.append(self.PLANT_MAP.get(self.rows[1][start_index], ""))
        if start_index + 1 < len(self.rows[1]):
            result.append(self.PLANT_MAP.get(self.rows[1][start_index + 1], ""))
            
        # Edge Case: Handle missing plants by filtering out empty strings
        return [plant for plant in result if plant]
    
    # Handled Edge Cases: empty diagram, diagram with only one row, diagram with more than two rows,
    #                     None students parameter, empty students list, mismatched row lengths,
    #                     student not in roster, empty student roster, index out of bounds for rows,
    #                     missing plants

    """

        'C': 'Clover',
        'G': 'Grass'
    }

        """

        if not diagram:
            self.rows = ["", ""]
        else:
            lines = diagram.split('

            # Edge Case: Handle diagram with only one row
            if len(lines) == 1:
                self.rows = [lines[0], ""]
            else:
                self.rows = [lines[0], lines[1]]

    """
    A class representing a garden with plants assigned to students.
    
    The garden consists of two rows of plants, and each student is responsible
    for four plants (two from each row). Students are assigned plants based
    on their alphabetical position.
    """
    
    PLANT_MAP = {
        'G': 'Grass',
        'C': 'Clover',
        'R': 'Radishes',
        'V': 'Violets'
    }
    
    DEFAULT_STUDENTS = [
        "Alice", "Bob", "Charlie", "David", "Eve", "Fred",
        "Ginny", "Harriet", "Ileana", "Joseph", "Kincaid", "Larry"
    ]
    def __init__(self, diagram: str, students: list[str] | None = None) -> None:
        """
        Initialize a Garden with a diagram and optional student list.
        
        Args:
            diagram: A string representing the garden layout with two rows
            students: Optional list of student names (defaults to standard 12 students)
        """
        # Edge Case: Handle empty diagram
        if not diagram:
            self.rows = ["", ""]
        else:
            rows = diagram.split('
')
            # Edge Case: Handle diagram with only one row
            if len(rows) == 1:
                self.rows = [rows[0], ""]
            # Edge Case: Handle diagram with more than two rows (take only first two)
            elif len(rows) >= 2:
                self.rows = [rows[0], rows[1]]
        
        # Edge Case: Handle students parameter being None
        if students is None:
            self.students = sorted(self.DEFAULT_STUDENTS)
        else:
            # Edge Case: Handle empty students list
            self.students = sorted(students) if students else []
        
        # Edge Case: Handle mismatched row lengths by taking minimum length
        if self.rows[0] and self.rows[1]:
            min_length = min(len(self.rows[0]), len(self.rows[1]))
            self.rows = [self.rows[0][:min_length], self.rows[1][:min_length]]
        # Edge Case: Handle one empty row
        elif not self.rows[0] and self.rows[1]:
            self.rows = ["", ""]
        elif self.rows[0] and not self.rows[1]:
            self.rows = ["", ""]
    def plants(self, student: str) -> list[str]:
        """
        Get the list of plants for a given student.
        
        Args:
            student: The name of the student
            
        Returns:
            A list of four plant names belonging to the student
            
        Raises:
            ValueError: If the student is not in the roster
        """
        # Edge Case: Handle student not in roster
        if student not in self.students:
            raise ValueError(f"Student {student} not found in roster")
        
        # Edge Case: Handle empty students list
        if not self.students:
            return []
        
        student_index = self.students.index(student)
        position = student_index * 2
        
        result = []
        
        # Edge Case: Handle position beyond row length

        else:

    A class representing a garden with plants assigned to students.
    
    The garden consists of two rows of plants, and each student is responsible
    for four plants (two from each row). Students are assigned plants based
    on their alphabetical position.
    """
    
    PLANT_CODES = {
        'V': 'Violets',
        'R': 'Radishes',
        'C': 'Clover',
        'G': 'Grass'
    }
    
    DEFAULT_STUDENTS = [
        "Alice", "Bob", "Charlie", "David", "Eve", "Fred",
        "Ginny", "Harriet", "Ileana", "Joseph", "Kincaid", "Larry"
    ]
    def __init__(self, diagram: str, students: list[str] | None = None) -> None:
        """
        Initialize a Garden with a diagram and optional student list.
        
        Args:
            diagram: A string representing the garden layout with two rows
            students: Optional list of student names (defaults to predefined list)
        """
        # Edge Case: Empty diagram
        if not diagram:
            self.rows = ["", ""]
        else:
            rows = diagram.split('
')
            # Edge Case: Diagram with only one row
            if len(rows) == 1:
                self.rows = [rows[0], ""]
            # Edge Case: Diagram with more than two rows (only use first two)
            elif len(rows) > 2:
                self.rows = rows[:2]
            else:
                self.rows = rows
        
        # Edge Case: Empty student list provided
        if students is not None and len(students) == 0:
            self.students = []
        else:
            self.students = sorted(students) if students is not None else self.DEFAULT_STUDENTS.copy()
        
        # Edge Case: Uneven row lengths (pad shorter row with empty spaces)
        if len(self.rows) >= 2:
            max_length = max(len(self.rows[0]), len(self.rows[1]))
            self.rows[0] = self.rows[0].ljust(max_length)
            self.rows[1] = self.rows[1].ljust(max_length)
        elif len(self.rows) == 1:
            self.rows.append("".ljust(len(self.rows[0])))
    def plants(self, student: str) -> list[str]:
        """
        Get the list of plants belonging to a specific student.

        # Edge Case: Student not in roster
        if student not in self.students:
            raise ValueError(f"Student '{student}' is not in the roster")

        if not self.rows or len(self.rows) < 2:
            return []
        
        student_index = self.students.index(student)
        position = student_index * 2
        
        # Edge Case: Student position exceeds available plants
        if position >= len(self.rows[0]) or position >= len(self.rows[1]):
            return []

        # Get plants from first row
        if position < len(self.rows[0]):
            plant1_code = self.rows[0][position]
            if plant1_code in self.PLANT_CODES:
                result.append(self.PLANT_CODES[plant1_code])

            plant2_code = self.rows[0][position + 1]
            if plant2_code in self.PLANT_CODES:
                result.append(self.PLANT_CODES[plant2_code])

            plant3_code = self.rows[1][position]

        # Invalid codes are simply ignored, so we might return fewer than 4 plants
        # This is handled by only adding plants with valid codes to the result

class Garden:
    """
    A class representing a garden with plants assigned to students.
    
    The garden consists of two rows of plants, and each student is responsible
    for four plants (two from each row). Students are assigned plants based
    on their alphabetical position.
    """
    
    PLANT_CODES = {
        'V': 'Violets',
        'R': 'Radishes',
        'C': 'Clover',
        'G': 'Grass'
    }
    
    DEFAULT_STUDENTS = [
        "Alice", "Bob", "Charlie", "David", "Eve", "Fred",
        "Ginny", "Harriet", "Ileana", "Joseph", "Kincaid", "Larry"
    ]
    def __init__(self, diagram: str, students: list[str] | None = None) -> None:
        """
        Initialize a Garden with a diagram and optional student list.
        
        Args:
            diagram: A string representing the garden layout with two rows
            students: Optional list of student names (defaults to standard 12 students)
        """
        # Edge Case: Handle empty diagram
        if not diagram:
            self.rows = ["", ""]
        else:
            lines = diagram.split('
')
            # Edge Case: Handle diagram with only one row
            if len(lines) == 1:
                self.rows = [lines[0], ""]
            # Edge Case: Handle diagram with more than two rows (use only first two)
            else:
                self.rows = lines[:2]
        
        # Edge Case: Handle None students parameter
        if students is None:
            self.students = sorted(self.DEFAULT_STUDENTS)
        else:
            # Edge Case: Handle empty students list
            self.students = sorted(students) if students else []
            
        # Edge Case: Handle student name duplication by using set for lookups
        self.student_indices = {name: i for i, name in enumerate(self.students)}
    def plants(self, student: str) -> list[str]:
        """
        Get the list of plants for a given student.
        
        Args:
            student: The name of the student
            
        Returns:
            A list of four plant names belonging to the student
            
        Raises:
            ValueError: If the student is not in the roster
        """
        # Edge Case: Handle student not in roster
        if student not in self.student_indices:
            raise ValueError(f"Student '{student}' not found in roster")
            
        student_index = self.student_indices[student]
        plant_start_index = student_index * 2
        
        result = []
        
        # Edge Case: Handle when rows are too short for student's position
        for row in self.rows:
            if plant_start_index < len(row):
                # Get first plant if it exists
                result.append(self.PLANT_CODES.get(row[plant_start_index], "Unknown"))
                # Get second plant if it exists
                if plant_start_index + 1 < len(row):
                    result.append(self.PLANT_CODES.get(row[plant_start_index + 1], "Unknown"))
                else:
                    # Edge Case: Handle missing second plant in row
                    result.append("Unknown")
            else:
                # Edge Case: Handle missing plants when row is too short
                result.extend(["Unknown", "Unknown"])
                
        return result

class Garden:
    """
    A class representing a garden with plants assigned to students.
    
    The garden consists of two rows of plants, and each student is responsible
    for four plants (two from each row). Students are assigned plants based
    on their position in an alphabetically sorted list.
    """
    
    PLANT_CODES = {
        'V': 'Violets',
        'R': 'Radishes',
        'C': 'Clover',
        'G': 'Grass'
    }
    
    DEFAULT_STUDENTS = [
        "Alice", "Bob", "Charlie", "David", "Eve", "Fred",
        "Ginny", "Harriet", "Ileana", "Joseph", "Kincaid", "Larry"
    ]
    def __init__(self, diagram: str, students: list[str] | None = None) -> None:
        """
        Initialize a Garden with a diagram and optional custom student list.
        
        Args:
            diagram: A string representing the garden layout with two rows
            students: Optional list of student names to override the default roster
        """
        # Edge Case: Handle empty diagram
        if not diagram:
            self.rows = ["", ""]
        else:
            rows = diagram.split('
')
            # Edge Case: Handle diagram with only one row
            if len(rows) == 1:
                self.rows = [rows[0], ""]
            # Edge Case: Handle diagram with more than two rows (take only first two)
            else:
                self.rows = rows[:2]
        
        # Edge Case: Handle None students parameter
        if students is None:
            self.students = sorted(self.DEFAULT_STUDENTS)
        else:
            # Edge Case: Handle empty students list
            self.students = sorted(students) if students else []
            
        # Edge Case: Handle student name duplication by using sorted unique names
        self.students = sorted(list(set(self.students)))
    def plants(self, student: str) -> list[str]:
        """
        Get the list of plants belonging to a specific student.
        
        Args:
            student: The name of the student
            
        Returns:
            A list of four plant names belonging to the student
            
        Raises:
            ValueError: If the student is not in the roster
        """
        # Edge Case: Handle student not in roster
        if student not in self.students:
            raise ValueError(f"Student '{student}' not found in roster")
            
        # Edge Case: Handle empty student name
        if not student:
            raise ValueError("Student name cannot be empty")
            
        student_index = self.students.index(student)
        position = student_index * 2
        
        result = []
        
        # Edge Case: Handle when position exceeds row length
        for row in self.rows:
            if position < len(row):
                # Edge Case: Handle when only one plant is available at position
                result.append(self.PLANT_CODES.get(row[position], "Unknown"))
                # Edge Case: Handle when position+1 exceeds row length
                if position + 1 < len(row):
                    result.append(self.PLANT_CODES.get(row[position + 1], "Unknown"))
                else:
                    result.append("Unknown")
            else:
                # Edge Case: Handle when position is beyond the row (fill with Unknown)
                result.extend(["Unknown", "Unknown"])
                
        # Edge Case: Handle invalid plant codes by filtering out Unknown values if needed
        # (In this implementation we keep them to maintain the 4-plant requirement)
        
        return result

class Garden:
    """
    A class representing a garden with plants assigned to students.
    
    The garden consists of two rows of plants, with each student being responsible
    for four plants (two from each row). Students are assigned plants based on
    their alphabetical position.
    """
    
    PLANT_MAP = {
        'G': 'Grass',
        'C': 'Clover',
        'R': 'Radishes',
        'V': 'Violets'
    }
    
    DEFAULT_STUDENTS = [
        "Alice", "Bob", "Charlie", "David", "Eve", "Fred",
        "Ginny", "Harriet", "Ileana", "Joseph", "Kincaid", "Larry"
    ]
    
    def __init__(self, diagram: str, students: list[str] | None = None) -> None:
        """
        Initialize a Garden with a diagram and optional student list.
        
        Args:
            diagram: A string representing the garden layout with two rows
            students: Optional list of student names to override default roster
        """
        # Edge Case: Handle empty diagram
        if not diagram:
            self.rows = ["", ""]
        else:
            rows = diagram.split('
')
            # Edge Case: Handle diagram with only one row
            if len(rows) == 1:
                self.rows = [rows[0], ""]
            # Edge Case: Handle diagram with more than two rows (take only first two)
            else:
                self.rows = rows[:2]
        
        # Edge Case: Handle empty student list by using default students
        if students is None:
            self.students = sorted(self.DEFAULT_STUDENTS)
        else:
            # Edge Case: Handle custom student list by sorting it
            self.students = sorted(students)
        
        # Edge Case: Handle rows of different lengths by taking minimum length
        if len(self.rows) >= 2:
            min_length = min(len(self.rows[0]), len(self.rows[1]))
            self.rows = [self.rows[0][:min_length], self.rows[1][:min_length]]
    
    def plants(self, student: str) -> list[str]:
        """
        Get the list of plants belonging to a specific student.
        
        Args:
            student: The name of the student
            
        Returns:
            A list of four plant names belonging to the student
            
        Raises:
            ValueError: If the student is not in the roster
        """
        # Edge Case: Handle student not in roster
        if student not in self.students:
            raise ValueError(f"Student '{student}' not found in roster")
        
        # Find student's position
        student_index = self.students.index(student)
        
        # Each student gets 2 consecutive plants from each row
        start_position = student_index * 2
        
        result = []
        
        # Edge Case: Handle when student position exceeds row length
        for row in self.rows:
            if start_position < len(row):
                # Get up to 2 plants from this row

                for i in range(plants_in_row):
                    plant_code = row[start_position + i]
                    # Edge Case: Handle invalid plant characters
                    if plant_code in self.PLANT_MAP:
                        result.append(self.PLANT_MAP[plant_code])
                    else:

                        result.append("Unknown")

// This is only a SKELETON file for the 'Point Mutations' exercise. It's been provided as a
// convenience to get you started writing code faster.

export class DNA {
  /**
   * @param {string} nucleotides
   */
  constructor(nucleotides) {
    this.nucleotides = nucleotides;
  }
  /**
   * @param {string} comparison
   * @returns {number}
   */
  hammingDistance(comparison) {
    // Edge Case: Empty strings
    if (this.nucleotides.length === 0 && comparison.length === 0) {
      return 0;
    }
    // Edge Case: Identical sequences
    if (this.nucleotides === comparison) {
      return 0;
    }
    // Edge Case: Single character sequences
    if (this.nucleotides.length === 1 && comparison.length === 1) {
      return this.nucleotides === comparison ? 0 : 1;
    }
    const minLength = Math.min(this.nucleotides.length, comparison.length);
    let distance = 0;
    for (let i = 0; i < minLength; i++) {
      if (this.nucleotides[i] !== comparison[i]) {
        distance++;
      }
    }
    return distance;
  }
}
// Handled Edge Cases: empty strings, identical sequences, single character sequences, sequences of different lengths

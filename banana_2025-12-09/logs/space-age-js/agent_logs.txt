/**
 * @param {string} planet
 * @param {number} seconds
 * @returns {number}
 */
export const age = (planet, seconds) => {
  // Edge Case: Negative or zero age in seconds
  if (seconds <= 0) {
    throw new Error('age must be positive');
  }
  // Define orbital periods in Earth years
  const orbitalPeriods = {
    'mercury': 0.2408467,
    'venus': 0.61519726,
    'earth': 1.0,
    'mars': 1.8808158,
    'jupiter': 11.862615,
    'saturn': 29.447498,
    'uranus': 84.016846,
    'neptune': 164.79132
  };
  // Edge Case: Invalid planet name input (case insensitive)
  const normalizedPlanet = planet.toLowerCase();
  if (!orbitalPeriods.hasOwnProperty(normalizedPlanet)) {
    throw new Error('not a planet');
  }
  // Earth year in seconds
  const earthYearInSeconds = 31557600;
  
  // Calculate age in Earth years
  const earthAge = seconds / earthYearInSeconds;
  
  // Calculate age on the specified planet
  const planetAge = earthAge / orbitalPeriods[normalizedPlanet];
  
  // Round to two decimal places
  return Math.round(planetAge * 100) / 100;
};
// Handled Edge Cases: Negative or zero age in seconds, Invalid planet name input (case insensitive)

/**
 * @param {string} planet
 * @param {number} seconds
 * @returns {number}
 */
export const age = (planet, seconds) => {
  // Edge Case: Negative or zero age in seconds
  if (seconds <= 0) {
    throw new Error('age must be positive');
  }
  // Define orbital periods in Earth years
  const orbitalPeriods = {
    'mercury': 0.2408467,
    'venus': 0.61519726,
    'earth': 1.0,
    'mars': 1.8808158,
    'jupiter': 11.862615,
    'saturn': 29.447498,
    'uranus': 84.016846,
    'neptune': 164.79132
  };
  // Edge Case: Invalid planet name input (case insensitive)
  const normalizedPlanet = planet.toLowerCase();
  if (!orbitalPeriods.hasOwnProperty(normalizedPlanet)) {
    throw new Error('not a planet');
  }
  // Calculate Earth years
  const earthYears = seconds / 31557600;
  
  // Calculate age on the specified planet
  const planetAge = earthYears / orbitalPeriods[normalizedPlanet];
  
  // Round to two decimal places
  return Math.round(planetAge * 100) / 100;
};
// Handled Edge Cases: Negative or zero age in seconds, Invalid planet name input (case insensitive)

/**
 * @param {string} planet
 * @param {number} seconds
 * @returns {number}
 */
export const age = (planet, seconds) => {
  // Edge Case: negative age in seconds
  if (seconds < 0) {
    throw new Error('age cannot be negative');
  }
  const earthYearInSeconds = 31557600;
  
  const planetOrbitalPeriods = {
    'mercury': 0.2408467,
    'venus': 0.61519726,
    'earth': 1.0,
    'mars': 1.8808158,
    'jupiter': 11.862615,
    'saturn': 29.447498,
    'uranus': 84.016846,
    'neptune': 164.79132
  };
  // Edge Case: invalid planet name (case insensitive)
  const normalizedPlanet = planet.toLowerCase();
  if (!planetOrbitalPeriods.hasOwnProperty(normalizedPlanet)) {
    throw new Error('not a planet');
  }
  // Edge Case: zero age in seconds
  if (seconds === 0) {
    return 0;
  }
  const orbitalPeriod = planetOrbitalPeriods[normalizedPlanet];
  const ageInEarthYears = seconds / earthYearInSeconds;
  const ageInPlanetYears = ageInEarthYears / orbitalPeriod;
  // Edge Case: floating point precision - round to 2 decimal places
  return Math.round(ageInPlanetYears * 100) / 100;
};

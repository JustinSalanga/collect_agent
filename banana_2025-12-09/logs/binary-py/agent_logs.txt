def parse_binary(binary_string: str) -> int:
    """
    Convert a binary string to its decimal equivalent using first principles.
    
    Args:
        binary_string: A string representing a binary number (e.g. '101010')
        
    Returns:
        int: The decimal equivalent of the binary number
        
    Raises:
        ValueError: If the input string contains invalid characters (not '0' or '1')
        
    Examples:
        >>> parse_binary('101')
        5
        >>> parse_binary('1000')
        8
        >>> parse_binary('0')
        0
        >>> parse_binary('')  # Edge Case: Empty string
        0
        >>> parse_binary('102')  # Edge Case: Invalid characters
        ValueError: Invalid binary literal: 102
    """
    # Edge Case: Empty string input
    if not binary_string:
        return 0
    
    # Edge Case: String containing characters other than '0' or '1'
    for char in binary_string:
        if char != '0' and char != '1':
            raise ValueError("Invalid binary literal: " + binary_string)
    
    result = 0
    # Process each digit from left to right
    for digit in binary_string:
        # Shift result left by 1 bit (multiply by 2) and add current digit
        result = result * 2 + int(digit)
    
    return result
    # Handled Edge Cases: Empty string input, String containing characters other than '0' or '1'

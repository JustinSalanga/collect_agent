    Attributes:

        """

            seconds (int): Age in seconds
            
        # Handled Edge Cases: negative seconds input
        """
        # Edge Case: negative seconds input
        if seconds < 0:
            raise ValueError("Seconds cannot be negative")
        
        self.seconds = seconds
        self.earth_year_in_seconds = 31557600  # 365.25 * 24 * 60 * 60

        Returns:

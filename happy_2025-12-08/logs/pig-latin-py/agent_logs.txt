def translate(text: str) -> str:
    if not text:
        return ""
    
    words = text.split()
    translated_words = []
    
    for word in words:
        # Edge Case: empty string
        if not word:
            translated_words.append("")
            continue
            
        # Edge Case: word with no alphabetic characters
        if not any(c.isalpha() for c in word):
            translated_words.append(word + "ay")
            continue
            
        # Find the index where the consonant cluster ends
        i = 0
        while i < len(word):
            # Edge Case: handle 'qu' as a unit
            if i < len(word) - 1 and word[i:i+2].lower() == 'qu':
                i += 2
                continue
            # Edge Case: 'y' is a vowel when it's not at the beginning of a word
            elif i > 0 and word[i].lower() == 'y':
                break
            # Edge Case: vowels
            elif word[i].lower() in 'aeiou':
                break
            else:
                i += 1
        
        # Edge Case: words starting with "xr" or "yt"
        if len(word) >= 2 and word[:2].lower() in ['xr', 'yt']:
            i = 0
            
        # Apply pig latin rules
        if i == 0:
            # Rule 1: starts with vowel or "xr"/"yt"
            pig_latin_word = word + "ay"
        else:
            # Rules 2, 3, 4: move consonants to the end
            pig_latin_word = word[i:] + word[:i] + "ay"
            
        translated_words.append(pig_latin_word)
    
    return " ".join(translated_words)

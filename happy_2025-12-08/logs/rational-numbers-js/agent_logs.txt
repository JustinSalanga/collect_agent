export class Rational {
  constructor(a, b) {
    if (b === 0) {
      throw new Error('Denominator cannot be zero');
    }
    
    this.numerator = a;
    this.denominator = b;
    this.reduce();
  }
  add(other) {
    // Edge Case: Adding two rational numbers
    const num = this.numerator * other.denominator + other.numerator * this.denominator;
    const den = this.denominator * other.denominator;
    return new Rational(num, den);
  }
  sub(other) {
    // Edge Case: Subtracting two rational numbers
    const num = this.numerator * other.denominator - other.numerator * this.denominator;
    const den = this.denominator * other.denominator;
    return new Rational(num, den);
  }
  mul(other) {
    // Edge Case: Multiplying two rational numbers
    const num = this.numerator * other.numerator;
    const den = this.denominator * other.denominator;
    return new Rational(num, den);
  }
  div(other) {
    // Edge Case: Dividing by zero rational number
    if (other.numerator === 0) {
      throw new Error('Cannot divide by zero');
    }
    
    // Edge Case: Dividing two rational numbers
    const num = this.numerator * other.denominator;
    const den = this.denominator * other.numerator;
    return new Rational(num, den);
  }
  abs() {
    // Edge Case: Absolute value of a rational number
    return new Rational(Math.abs(this.numerator), Math.abs(this.denominator));
  }
  exprational(n) {
    // Edge Case: Raising to power 0
    if (n === 0) {
      return new Rational(1, 1);
    }
    
    // Edge Case: Raising to a negative power
    if (n < 0) {
      const absN = Math.abs(n);
      return new Rational(
        Math.pow(this.denominator, absN),
        Math.pow(this.numerator, absN)
      );
    }
    
    // Edge Case: Raising to a positive power
    return new Rational(
      Math.pow(this.numerator, n),
      Math.pow(this.denominator, n)
    );
  }
  expreal(base) {
    // Edge Case: Raising a real number to a rational power
    return Math.pow(base, this.numerator / this.denominator);
  }
  reduce() {
    // Edge Case: Reducing to lowest terms with negative denominator
    if (this.denominator < 0) {
      this.numerator *= -1;
      this.denominator *= -1;
    }
    
    // Edge Case: Reducing to lowest terms
    const gcd = this.gcd(Math.abs(this.numerator), Math.abs(this.denominator));
    this.numerator /= gcd;
    this.denominator /= gcd;
    
    return this;
  }
  gcd(a, b) {
    // Edge Case: Calculating GCD with zero values
    if (b === 0) {
      return a;
    }
    // Prevent potential infinite recursion by ensuring progress toward base case
    if (a < b) {
      return this.gcd(b, a);
    }
    return this.gcd(b, a % b);
  }

+      this.denominator *= -1;
+    }

+    const gcd = this.gcd(Math.abs(this.numerator), Math.abs(this.denominator));
+    this.numerator /= gcd;
+    this.denominator /= gcd;
+    
+    return this;
   }

+  gcd(a, b) {
+    // Edge Case: Calculating GCD with zero values
+    if (b === 0) {
+      return a;
+    }
+    // Prevent potential infinite recursion by ensuring progress toward base case
+    if (a < b) {
+      return this.gcd(b, a);
+    }
+    return this.gcd(b, a % b);
+  }
